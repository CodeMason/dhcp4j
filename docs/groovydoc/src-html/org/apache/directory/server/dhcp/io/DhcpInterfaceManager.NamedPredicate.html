034/**
035 * A utility base class for DHCP server message handlers which identifies
036 * the interface on which a packet was received.
037 * 
038 * <ol>
039 * <li>Packet is received by the message handler and calls {@link #newRequestContext}.</li>
040 * <li>This class guesses which interface(s) the packet may have come on.</li>
041 * <li>This class constructs a {@link DhcpRequestContext} containing that information.</li>
042 * <li>The user's {@link LeaseManager} eventually calls {@link DhcpRequestContext#setClientAddress(InetAddress)}.</li>
043 * <li>The DhcpRequestContext class refines the interface guess and permits calls to {@link DhcpRequestContext#getInterfaceAddress()}.</li>
044 * <li>The message handler uses {@link DhcpRequestContext#getInterfaceAddress()} to update and send the new packet.</li>
045 * </ol>
046 *
047 * This base class helps to work around various weaknesses with broadcast UDP in Java.
048 *
049 * @author shevek
050 */
051public class DhcpInterfaceManager {
052
053    private static final Logger LOG = LoggerFactory.getLogger(DhcpInterfaceManager.class);
054
055    public static class ValidPredicate implements Predicate<NetworkInterface> {
056
057        @Override
058        public boolean apply(NetworkInterface iface) {
059            try {
060                if (iface == null) {
061                    LOG.debug("Ignoring NetworkInterface: null!");
062                    return false;
063                }
064                // Bridges claim to be down when no attached interfaces are up. :-(
065                // if (false && !iface.isUp()) { LOG.debug("Ignoring NetworkInterface: Down: {}", iface); return false; }
066                if (iface.isLoopback()) {
067                    LOG.debug("Ignoring NetworkInterface: Loopback: {}", iface);
068                    return false;
069                }
070                if (iface.isPointToPoint()) {
071                    LOG.debug("Ignoring NetworkInterface: PointToPoint: {}", iface);
072                    return false;
073                }
074                return true;
075            } catch (SocketException e) {
076                LOG.error("Failed to query " + iface, e);
077                return false;
078            }
079        }
080    }
081
082    public static class NamedPredicate extends ValidPredicate {
083
084        private final List<? extends String> names;
085
086        public NamedPredicate(@Nonnull List<? extends String> names) {
087            this.names = names;
088        }
089
090        public NamedPredicate(@Nonnull Iterable<? extends String> names) {
091            this(Lists.newArrayList(names));
092        }
093
094        public NamedPredicate(@Nonnull String... names) {
095            this(Arrays.asList(names));
096        }
097
098        @Override
099        public boolean apply(NetworkInterface iface) {
100            String name = iface.getName();
101            if (!names.contains(name)) {
102                LOG.debug("Ignoring NetworkInterface: Not included by name: {} in {}", name, names);
103                return false;
104            }
105            return super.apply(iface);
106        }
107    }

126    /**
127     * If this returns an address, it is NOT the global address.
128     */
129    @CheckForNull
130    public InterfaceAddress getResponseInterface(@Nonnull Object... objects) throws DhcpException {
131        InetAddress address = DhcpInterfaceUtils.toInetAddress(objects);
132        if (address != null) {
133            InterfaceAddress iface = getInterface(address);
134            if (iface != null)
135                return iface;
136        }
137        return null;
138    }
139
140    /**
141     * Returns an object which may be used to later decide on a specific server interface.
142     */
143    @CheckForNull
144    public DhcpRequestContext newRequestContext(
145            @Nonnull InetSocketAddress serviceAddress,
146            @Nonnull InetSocketAddress remoteAddress,
147            @Nonnull InetSocketAddress localAddress,
148            @Nonnull DhcpMessage request) throws DhcpException {
149        InetAddress address = DhcpInterfaceUtils.toInetAddress(request.getRelayAgentAddress(), localAddress, serviceAddress, remoteAddress, request);
150        if (address == null)
151            // We don't know the remote system's address. Let's see if getting a lease helps us at all.
152            return new DhcpRequestContext(interfaces.keySet(), remoteAddress, localAddress);
153        InterfaceAddress interfaceAddress = getInterface(address);
154        if (interfaceAddress == null)
155            // We know the remote system's address but we don't think we can talk to it.
156            return null;
157        // We know the remote system's address. The lease had better match.
158        return new DhcpRequestContext(interfaceAddress, remoteAddress, localAddress);
159    }
160
161    @CheckForNull
162    public InterfaceAddress getInterface(@Nonnull InetAddress address) {
163        Preconditions.checkNotNull(address, "Address was null.");
164        for (InterfaceAddress iface : interfaces.keySet()) {
165            if (iface.isLocal(address))
166                return iface;
167        }
168        return null;
169    }
170
171    public void addInterface(@Nonnull InterfaceAddress address) throws IOException, InterruptedException {
172        LOG.debug("Adding InterfaceAddress: {}", address);
173        interfaces.put(address, Dummy.INSTANCE);
174    }
175
176    public void addInterface(@Nonnull NetworkInterface iface) throws IOException, InterruptedException {
177        for (java.net.InterfaceAddress address : iface.getInterfaceAddresses()) {
178            if (!AddressUtils.isUnicastAddress(address.getAddress())) {
179                LOG.debug("Ignoring InterfaceAddress: Not unicast: {}", address);
180                continue;
181            }
182            if (!(address.getAddress() instanceof Inet4Address)) {
183                LOG.debug("Ignoring InterfaceAddress: Not IPv4: {}", address);
184                continue;
185            }
186
187            addInterface(new InterfaceAddress(address.getAddress(), address.getNetworkPrefixLength()));
188        }
189    }
190
191    private void addInterfaces(@Nonnull Enumeration<? extends NetworkInterface> ifaces, @Nonnull Predicate<? super NetworkInterface> predicate) throws IOException, InterruptedException {
192        for (NetworkInterface iface : Collections.list(ifaces)) {
193            if (predicate.apply(iface)) {
194                LOG.debug("Adding NetworkInterface: {}", iface);
195                addInterface(iface);
196            }
197            addInterfaces(iface.getSubInterfaces(), predicate);
198        }
199    }
200
201    public void addInterfaces(@Nonnull Predicate<? super NetworkInterface> predicate) throws IOException, InterruptedException {
202        addInterfaces(NetworkInterface.getNetworkInterfaces(), predicate);
203    }
204
205    public void addDefaultInterfaces() throws IOException, InterruptedException {
206        addInterfaces(new ValidPredicate());
