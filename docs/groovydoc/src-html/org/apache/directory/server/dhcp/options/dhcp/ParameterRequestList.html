020package org.apache.directory.server.dhcp.options.dhcp;
021
022import org.apache.directory.server.dhcp.DhcpException;
023import org.apache.directory.server.dhcp.options.DhcpOption;
024import org.apache.directory.server.dhcp.options.DhcpOptionsRegistry;
025
026/**
027 * This option is used by a DHCP client to request values for specified
028 * configuration parameters.  The list of requested parameters is
029 * specified as n octets, where each octet is a valid DHCP option code
030 * as defined in this document.
031 * 
032 * The client MAY list the options in order of preference.  The DHCP
033 * server is not required to return the options in the requested order,
034 * but MUST try to insert the requested options in the order requested
035 * by the client.
036 * 
037 * The code for this option is 55.  Its minimum length is 1.
038 * 
039 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
040 */
041public class ParameterRequestList extends DhcpOption {
042    /*
043     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
044     */
045
046    @Override
047    public byte getTag() {
048        return 55;
049    }
050
051    @Override
052    protected String toStringData() throws DhcpException {
053        DhcpOptionsRegistry registry = DhcpOptionsRegistry.getInstance();
054        StringBuilder buf = new StringBuilder("[ ");
055        for (byte tag : getData()) {
056            Class<? extends DhcpOption> type = registry.getOptionType(tag);
057            if (type == null)
058                type = UnrecognizedOption.class;
059            buf.append(type.getSimpleName());
060            buf.append("(").append(tag & 0xFF).append(") ");
061        }
062        buf.append("]");
063        return buf.toString();
064    }
065
066}
