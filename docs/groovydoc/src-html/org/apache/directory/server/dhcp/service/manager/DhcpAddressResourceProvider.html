005package org.apache.directory.server.dhcp.service.manager;
006
007import com.google.common.base.Objects;
008import com.google.common.base.Preconditions;
009import com.google.common.net.InetAddresses;
010import java.net.InetAddress;
011import java.util.Arrays;
012import javax.annotation.Nonnull;
013import org.anarres.dhcp.common.address.AddressUtils;
014import org.anarres.dhcp.common.address.Subnet;
015import org.anarres.jallocator.ResourceProvider;
016
017/**
018 *
019 * @author shevek
020 */
021public class DhcpAddressResourceProvider implements ResourceProvider<InetAddress> {
022
023    private final byte[] rangeStartBytes;
024    private final long rangeSize;
025
026    // private final byte[] addressBytes = new byte[rangeStartBytes.length];
027    public DhcpAddressResourceProvider(@Nonnull Subnet subnet) {
028        Preconditions.checkNotNull(subnet, "Subnet was null.");
029        rangeStartBytes = subnet.getRangeStart().getAddress();
030        rangeSize = subnet.getRangeSize();
031    }
032
033    @Override
034    public long getResourceCount() {
035        return rangeSize;
036    }
037
038    @Override
039    public InetAddress getResource(long index) {
040        byte[] addressBytes = Arrays.copyOf(rangeStartBytes, rangeStartBytes.length);
041        addressBytes = AddressUtils.add(addressBytes, index);
042        return AddressUtils.toInetAddress(addressBytes);
043    }
044
045    @Override
046    public int hashCode() {
047        return Arrays.hashCode(rangeStartBytes) ^ (int) rangeSize;
048    }
049
050    @Override
051    public boolean equals(Object obj) {
052        if (this == obj)
053            return true;
054        if (null == obj)
055            return false;
056        if (getClass() != obj.getClass())
057            return false;
058        DhcpAddressResourceProvider other = (DhcpAddressResourceProvider) obj;
059        return Arrays.equals(rangeStartBytes, other.rangeStartBytes) && rangeSize == other.rangeSize;
060    }
061
062    @Override
063    public String toString() {
064        InetAddress rangeStart = AddressUtils.toInetAddress(rangeStartBytes);
065        InetAddress rangeEnd = getResource(getResourceCount() - 1);
066        return Objects.toStringHelper(this)
067                .add("rangeStartBytes", Arrays.toString(rangeStartBytes))
068                .add("rangeStart", InetAddresses.toAddrString(rangeStart))
069                .add("rangeEnd", InetAddresses.toAddrString(rangeEnd))
070                .add("rangeSize", rangeSize)
071                .toString();
072    }
073}
