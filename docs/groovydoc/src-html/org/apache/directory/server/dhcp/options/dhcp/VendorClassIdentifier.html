020package org.apache.directory.server.dhcp.options.dhcp;
021
022import org.apache.directory.server.dhcp.options.StringOption;
023
024/**
025 * This option is used by DHCP clients to optionally identify the vendor
026 * type and configuration of a DHCP client.  The information is a string
027 * of n octets, interpreted by servers.  Vendors may choose to define
028 * specific vendor class identifiers to convey particular configuration
029 * or other identification information about a client.  For example, the
030 * identifier may encode the client's hardware configuration.  Servers
031 * not equipped to interpret the class-specific information sent by a
032 * client MUST ignore it (although it may be reported). Servers that
033 * 
034 * respond SHOULD only use option 43 to return the vendor-specific
035 * information to the client.
036 * 
037 * The code for this option is 60, and its minimum length is 1.
038 * 
039 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
040 */
041public class VendorClassIdentifier extends StringOption {
042    /*
043     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
044     */
045
046    @Override
047    public byte getTag() {
048        return 60;
049    }
050
051}
