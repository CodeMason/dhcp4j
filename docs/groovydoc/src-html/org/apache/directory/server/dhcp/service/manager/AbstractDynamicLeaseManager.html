029/**
030 * A base class for lease managers which allocate addresses from a pool.
031 *
032 * @author shevek
033 */
034public abstract class AbstractDynamicLeaseManager extends AbstractLeaseManager {
035
036    private static final Logger LOG = LoggerFactory.getLogger(AbstractDynamicLeaseManager.class);
037    public static final int TTL_EXPIRE = 600;
038    private final Cache<NetworkAddress, ResourceAllocator<InetAddress>> allocators = CacheBuilder.newBuilder()
039            .expireAfterAccess(TTL_LEASE.maxLeaseTime * 4, TimeUnit.SECONDS)
040            .recordStats()
041            .build();
042
043    @CheckForNull
044    protected abstract InetAddress getFixedAddressFor(@Nonnull HardwareAddress hardwareAddress) throws DhcpException;
045
046    @CheckForNull
047    protected abstract Subnet getSubnetFor(@Nonnull NetworkAddress networkAddress) throws DhcpException;
048
049    @CheckForNull
050    private ResourceAllocator<InetAddress> getAllocatorFor(@Nonnull NetworkAddress networkAddress) throws DhcpException {
051        final Subnet subnet = getSubnetFor(networkAddress);
052        if (subnet == null)
053            return null;
054        NetworkAddress network = subnet.getNetworkAddress();
055        ResourceAllocator<InetAddress> allocator = allocators.getIfPresent(network);
056        // TODO: Update the resource allocator if it's out of date.
057        if (allocator != null)
058            return allocator;
059        try {
060            return allocators.get(network, new Callable<ResourceAllocator<InetAddress>>() {
061                @Override
062                public ResourceAllocator<InetAddress> call() throws Exception {
063                    DhcpAddressResourceProvider resourceProvider = new DhcpAddressResourceProvider(subnet);
064                    return new ResourceAllocator<InetAddress>(resourceProvider);
065                }
066            });
067        } catch (ExecutionException e) {
068            throw new DhcpException("Failed to load ResourceAllocator for " + subnet, e);
069        }
070    }
071
072    /**
073     * Leases an InetAddress for the given HardwareAddress.
074     *
075     * Lock, retrieve current mapping from store.
076     * If the InetAddress is unallocated OR allocated to the given HardwareAddress, return it.
077     * Else return null.
078     */
079    @CheckForNull
080    protected abstract boolean leaseIp(
081            @Nonnull InetAddress address,
082            @Nonnull HardwareAddress hardwareAddress,
083            @Nonnegative long ttl) throws Exception;
084
085    @CheckForNull
086    protected InetAddress leaseMac(
087            @Nonnull DhcpRequestContext context,
088            @Nonnull HardwareAddress hardwareAddress,
089            @CheckForNull InetAddress currentAddress, @CheckForNull InetAddress requestedAddress,
090            @Nonnegative long ttl)
091            throws Exception {
092
093        // Is it a singular system?
094        // If it's singular, perhaps it deliberately gets no DHCP.
095        // What network should the system address come from?
096        // String hardwareId = InventoryUtils.hardwareAddress(hardwareAddress);
097        // UUID systemId = new InventoryManager.GetSystemIdCommand(inventoryManager, hardwareId).execute();
098        // BareMetalBootParameters bootParameters = new BareMetalManager.GetBootParametersCommand(bareMetalManager, systemId).execute();
099        // Fixed address from configuration.
100        FIXED:
101        {
102            LOG.debug("Looking for fixed address.");
103            InetAddress fixedAddress = getFixedAddressFor(hardwareAddress);
104            if (fixedAddress != null) {
105                if (!leaseIp(fixedAddress, hardwareAddress, ttl * 2))
106                    LOG.error("Client configured with fixed-address " + fixedAddress + " but lease failed.");
107                else if (LOG.isDebugEnabled())
108                    LOG.debug("Using fixed address " + fixedAddress);
109                return fixedAddress;
110            }
111        }
112
113        // Current or previous lease address, if available.
114        EXISTING:
115        {
116            LOG.debug("Looking for pre-existing address.");
117            if (currentAddress != null) {
118                if (leaseIp(currentAddress, hardwareAddress, ttl * 2)) {
119                    if (LOG.isDebugEnabled())
120                        LOG.debug("Using pre-existing address " + currentAddress);
121                    return currentAddress;
122                }
123            }
124        }
125
126        REQUESTED:
127        {
128            LOG.debug("Looking for client-requested address.");
129            // Requested address, if available.
130            if (requestedAddress != null) {
131                if (leaseIp(requestedAddress, hardwareAddress, ttl * 2)) {
132                    if (LOG.isDebugEnabled())
133                        LOG.debug("Using client-requested address {}", requestedAddress);
134                    return requestedAddress;
135                }
136            }
137        }
138
139        CREATED:
140        for (InterfaceAddress interfaceAddress : context.getInterfaceAddresses()) {
141            // New address from the server's pool
142            ResourceAllocator<InetAddress> allocator = getAllocatorFor(interfaceAddress.toNetworkAddress());
143            if (allocator == null) {
144                if (LOG.isDebugEnabled())
145                    LOG.debug("No address allocator for {}", interfaceAddress);
146                continue CREATED;
147            }
148            for (InetAddress allocatedAddress : allocator) {
149                if (leaseIp(allocatedAddress, hardwareAddress, ttl * 2)) {
150                    if (LOG.isDebugEnabled())
151                        LOG.debug("Using server-generated address {}", allocatedAddress);
152                    return allocatedAddress;
153                }
154            }
155        }
156        LOG.warn("Failed to generate a lease for {}", hardwareAddress);
157
158        return null;
159    }
160
161    /**
162     * Leases an InetAddress for the given HardwareAddress.
163     *
164     * Lock, retrieve current mapping from store.
165     * Call {@link #leaseMac(DhcpRequestContext, HardwareAddress, InetAddress, InetAddress, long)}
166     * with the current mapping.
167     * Store the returned address (if given) as the new current mapping and release the lock.
168     */
169    @CheckForNull
170    protected abstract InetAddress leaseMac(
171            @Nonnull DhcpRequestContext context,
172            @Nonnull DhcpMessage request,
173            @CheckForNull InetAddress clientRequestedAddress, @Nonnegative long ttl)
174            throws Exception;
175
176    @Override
177    public DhcpMessage leaseOffer(
178            DhcpRequestContext context,
179            DhcpMessage request,
180            InetAddress clientRequestedAddress, long clientRequestedExpirySecs)
181            throws DhcpException {
182        // LOG.info("OFFER: interfaceAddress=" + interfaceAddress + ", networkAddress=" + networkAddress + ", request=" + request + ", hardwareAddress=" + hardwareAddress + ", requestedAddress=" + requestedAddress + ", requestedLeaseTimeSecs=" + requestedLeaseTimeSecs);
183        try {
184            long leaseTimeSecs = getLeaseTime(TTL_OFFER, clientRequestedExpirySecs);
185            InetAddress clientAddress = leaseMac(context, request, clientRequestedAddress, leaseTimeSecs);
186            if (clientAddress == null)
187                return null;
188            return newReplyAck(request, MessageType.DHCPACK, clientAddress, leaseTimeSecs);
189        } catch (Exception e) {
190            Throwables.propagateIfPossible(e, DhcpException.class);
191            throw new DhcpException("Failed to lease for MAC " + request.getHardwareAddress() + ": " + e, e);
192        }
193    }
194
195    @Override
196    public DhcpMessage leaseRequest(
197            DhcpRequestContext context,
198            DhcpMessage request,
199            InetAddress clientRequestedAddress, long clientRequestedExpirySecs)
200            throws DhcpException {
201        try {
202            long leaseTimeSecs = getLeaseTime(TTL_LEASE, clientRequestedExpirySecs);
203            if (clientRequestedAddress == null) {
204                LOG.warn("REQUEST from " + request.getHardwareAddress() + " did not request an address.");
205                return null;
206            }
207            InetAddress clientAddress = leaseMac(context, request, clientRequestedAddress, leaseTimeSecs);
208            if (clientAddress == null)
209                return null;
210            return newReplyAck(request, MessageType.DHCPACK, clientAddress, leaseTimeSecs);
211        } catch (Exception e) {
212            Throwables.propagateIfPossible(e, DhcpException.class);
213            throw new DhcpException("Failed to lease for MAC " + request.getHardwareAddress() + ": " + e, e);
214        }
215    }
