006package org.apache.directory.server.dhcp.service.store;
007
008import java.net.InetAddress;
009import java.net.UnknownHostException;
010import java.util.Hashtable;
011import javax.annotation.CheckForNull;
012import javax.annotation.Nonnegative;
013import javax.annotation.Nonnull;
014import javax.naming.Context;
015import javax.naming.NamingEnumeration;
016import javax.naming.NamingException;
017import javax.naming.directory.Attribute;
018import javax.naming.directory.Attributes;
019import javax.naming.directory.InitialDirContext;
020import javax.naming.directory.SearchControls;
021import javax.naming.directory.SearchResult;
022import org.anarres.dhcp.common.address.AddressUtils;
023import org.apache.directory.server.dhcp.DhcpException;
024import org.apache.directory.server.dhcp.io.DhcpRequestContext;
025import org.apache.directory.server.dhcp.messages.DhcpMessage;
026import org.apache.directory.server.dhcp.messages.HardwareAddress;
027import org.apache.directory.server.dhcp.messages.MessageType;
028import org.apache.directory.server.dhcp.options.vendor.HostName;
029import org.apache.directory.server.dhcp.service.manager.AbstractLeaseManager;
030
031/**
032 *
033 * @author shevek
034 */
035public class LdapStoreLeaseManager extends AbstractLeaseManager {
036
037    private final InitialDirContext context;
038
039    public LdapStoreLeaseManager() throws NamingException {
040        Hashtable<String, Object> env = new Hashtable<String, Object>();
041        env.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
042        // env.put(Context.INITIAL_CONTEXT_FACTORY, DEFAULT_INITIAL_CONTEXT_FACTORY);
043        env.put(Context.PROVIDER_URL, "ldap://localhost:389/dc=tcat,dc=test");
044        this.context = new InitialDirContext(env);
045    }
046
047    /**
048     * @param hardwareAddress
049     * @return Host
050     * @throws DhcpException
051     */
052    @CheckForNull
053    protected DhcpConfigHost getHost(@Nonnull HardwareAddress hardwareAddress) throws DhcpException {
054        try {
055            String filter = "(&(objectclass=ipHost)(objectclass=ieee802Device)(macaddress={0}))";
056            SearchControls sc = new SearchControls();
057            sc.setCountLimit(1);
058            sc.setSearchScope(SearchControls.SUBTREE_SCOPE);
059            NamingEnumeration<SearchResult> ne = context.search("", filter, new Object[]{hardwareAddress.toString()}, sc);
060
061            if (ne.hasMoreElements()) {
062                SearchResult sr = ne.next();
063                Attributes att = sr.getAttributes();
064                Attribute ipHostNumberAttribute = att.get("iphostnumber");
065
066                if (ipHostNumberAttribute != null) {
067                    InetAddress clientAddress = InetAddress.getByName((String) ipHostNumberAttribute.get());
068                    Attribute cnAttribute = att.get("cn");
069
070                    return new DhcpConfigHost(
071                            cnAttribute != null ? (String) cnAttribute.get() : "unknown",
072                            hardwareAddress,
073                            clientAddress);
074                }
075            }
076        } catch (NamingException e) {
077            throw new DhcpException("Can't lookup lease", e);
078        } catch (UnknownHostException e) {
079            throw new DhcpException("Can't lookup lease", e);
080        }
081
082        return null;
083    }
084
085    @Nonnull
086    protected DhcpMessage newReply(
087            @Nonnull DhcpMessage request,
088            @Nonnull MessageType type,
089            @Nonnull DhcpConfigHost host,
090            @Nonnegative long leaseTimeSecs
091    ) {
092        DhcpMessage reply = newReplyAck(request, type, host.getClientAddress(), leaseTimeSecs);
093        reply.getOptions().add(new HostName(host.getName()));
094        reply.getOptions().addAll(host.getOptions());
095        return reply;
096    }
097
098    @Override
099    public DhcpMessage leaseOffer(
100            DhcpRequestContext context,
101            DhcpMessage request,
102            InetAddress clientRequestedAddress, long clientRequestedExpirySecs) throws DhcpException {
103        DhcpConfigHost host = getHost(request.getHardwareAddress());
104        if (host == null)
105            return null;
106        if (AddressUtils.isZeroAddress(host.getClientAddress()))
107            return null;
108        long leaseTimeSecs = getLeaseTime(TTL_OFFER, clientRequestedExpirySecs);
109        return newReply(request, MessageType.DHCPACK, host, leaseTimeSecs);
110    }
111
112    @Override
113    public DhcpMessage leaseRequest(
114            DhcpRequestContext context,
115            DhcpMessage request,
116            InetAddress clientRequestedAddress, long clientRequestedExpirySecs) throws DhcpException {
117        DhcpConfigHost host = getHost(request.getHardwareAddress());
118        if (host == null)
119            return null;
120        if (AddressUtils.isZeroAddress(host.getClientAddress()))
121            return null;
122        long leaseTimeSecs = getLeaseTime(TTL_LEASE, clientRequestedExpirySecs);
123        return newReply(request, MessageType.DHCPACK, host, leaseTimeSecs);
124    }
125}
