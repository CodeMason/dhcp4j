033    /** If this returns an address, it is NOT a zero address. */
034    @CheckForNull
035    private static InetAddress toInetAddress(@Nonnull DhcpMessage message) throws DhcpException {
036        // From the packet:
037        InetAddress assignedClientAddress = message.getAssignedClientAddress();
038        if (!AddressUtils.isZeroAddress(assignedClientAddress))
039            return assignedClientAddress;
040        InetAddress currentClientAddress = message.getCurrentClientAddress();
041        if (!AddressUtils.isZeroAddress(currentClientAddress))
042            return currentClientAddress;
043        // TODO: Should this catch locally?
044        InetAddress requestedClientAddress = message.getOptions().getAddressOption(RequestedIpAddress.class);
045        if (!AddressUtils.isZeroAddress(requestedClientAddress))
046            return requestedClientAddress;
047        InetAddress serverAddress = message.getOptions().getAddressOption(ServerIdentifier.class);
048        if (!AddressUtils.isZeroAddress(serverAddress))
049            return serverAddress;
050        InetAddress relayAgentAddress = message.getRelayAgentAddress();
051        if (!AddressUtils.isZeroAddress(relayAgentAddress))
052            return relayAgentAddress;
053        return null;
054    }
055
056    /** If this returns an address, it is NOT a zero address. */
057    @CheckForNull
058    private static InetAddress toInetAddress(@Nonnull SocketAddress sa) {
059        if (!(sa instanceof InetSocketAddress))
060            return null;
061        InetSocketAddress isa = (InetSocketAddress) sa;
062        InetAddress ia = isa.getAddress();
063        if (!AddressUtils.isZeroAddress(ia))
064            return ia;
065        return null;
066    }
067
068    /** If this returns an address, it is NOT a zero address. */
069    @CheckForNull
070    public static InetAddress toInetAddress(@Nonnull Object... objects) throws DhcpException {
071        for (Object object : objects) {
072            if (object == null)
073                continue;
074            if (object instanceof InetAddress) {
075                InetAddress a = (InetAddress) object;
076                if (!AddressUtils.isZeroAddress(a))
077                    return a;
078            } else if (object instanceof SocketAddress) {
079                InetAddress a = toInetAddress((SocketAddress) object);
080                if (a != null)
081                    return a;
082            } else if (object instanceof DhcpMessage) {
083                InetAddress a = toInetAddress((DhcpMessage) object);
084                if (a != null)
085                    return a;
086            } else {
087                throw new IllegalArgumentException("Cannot derive an InetAddress from " + object.getClass());
088            }
089        }
090        // LOG.debug("Failed to determine message address from " + Arrays.toString(objects));
091        return null;
092    }

094    /**
095     * Determine where to send the message: <br>
096     * If the 'giaddr' field in a DHCP message from a client is non-zero, the
097     * server sends any return messages to the 'DHCP server' port on the BOOTP
098     * relay agent whose address appears in 'giaddr'. If the 'giaddr' field is
099     * zero and the 'ciaddr' field is nonzero, then the server unicasts DHCPOFFER
100     * and DHCPACK messages to the address in 'ciaddr'. If 'giaddr' is zero and
101     * 'ciaddr' is zero, and the broadcast bit is set, then the server broadcasts
102     * DHCPOFFER and DHCPACK messages to 0xffffffff. If the broadcast bit is not
103     * set and 'giaddr' is zero and 'ciaddr' is zero, then the server unicasts
104     * DHCPOFFER and DHCPACK messages to the client's hardware address and
105     * 'yiaddr' address. In all cases, when 'giaddr' is zero, the server
106     * broadcasts any DHCPNAK messages to 0xffffffff.
107     */
108    @Nonnull
109    //This will suppress PMD.AvoidUsingHardCodedIP warnings in this class
110    @SuppressWarnings("PMD.AvoidUsingHardCodedIP")
111    public static InetSocketAddress determineMessageDestination(
112            @Nonnull DhcpMessage request,
113            @Nonnull DhcpMessage reply,
114            @Nonnull InterfaceAddress localAddress,
115            @Nonnegative int remotePort
116    ) {
117        if (!AddressUtils.isZeroAddress(request.getRelayAgentAddress())) {
118            // send to agent, if received via agent.
119            return new InetSocketAddress(request.getRelayAgentAddress(), DhcpService.SERVER_PORT);
120        } else if (reply.getMessageType() == MessageType.DHCPNAK) {
121            // force broadcast for DHCPNAKs
122            return new InetSocketAddress(localAddress.getBroadcastAddress(), remotePort);
123        } else if (!AddressUtils.isZeroAddress(request.getCurrentClientAddress())) {
124            // have a current address? unicast to it.
125            return new InetSocketAddress(request.getCurrentClientAddress(), remotePort);
126        } else {
127            // not a NAK...
128            return new InetSocketAddress(localAddress.getBroadcastAddress(), remotePort);
129        }
130    }
