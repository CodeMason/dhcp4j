020package org.apache.directory.server.dhcp.options.misc;
021
022import org.apache.directory.server.dhcp.options.ByteOption;
023
024/**
025 * The NetBIOS node type option allows NetBIOS over TCP/IP clients which
026 * are configurable to be configured as described in RFC 1001/1002.  The
027 * value is specified as a single octet which identifies the client type
028 * as follows:
029 * 
030 *    Value         Node Type
031 *    -----         ---------
032 *    0x1           B-node
033 *    0x2           P-node
034 *    0x4           M-node
035 *    0x8           H-node
036 * 
037 * In the above chart, the notation '0x' indicates a number in base-16
038 * (hexadecimal). The code for this option is 46. The length of this option is
039 * always 1.
040 */
041public class NetbiosNodeType extends ByteOption {
042    /*
043     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
044     */
045
046    @Override
047    public byte getTag() {
048        return 46;
049    }
050}
