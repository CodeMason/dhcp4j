044/**
045 * Abstract implementation of the server-side DHCP protocol. This class just
046 * provides some utility methods and dispatches server-bound messages to handler
047 * methods which can be overridden to provide the functionality.
048 * <p>
049 * Client-bound messages and BOOTP messages are ignored.
050 *
051 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
052 *
053 */
054public abstract class AbstractDhcpService extends AbstractDhcpReplyFactory implements DhcpService {
055
056    private static final Logger LOG = LoggerFactory.getLogger(AbstractDhcpService.class);
057
058    @Override
059    public DhcpMessage getReplyFor(
060            DhcpRequestContext context,
061            DhcpMessage request)
062            throws DhcpException {
063        Preconditions.checkNotNull(context, "DhcpRequestContext was null.");
064
065        // ignore messages with an op != REQUEST/REPLY
066        if ((request.getOp() != DhcpMessage.OP_BOOTREQUEST)
067                && (request.getOp() != DhcpMessage.OP_BOOTREPLY)) {
068            LOG.warn("Request operator is not BOOTREQUEST or BOOTREPLY: " + request);
069            return null;
070        }
071
072        // message type option MUST be set - we don't support plain BOOTP.
073        if (request.getMessageType() == null) {
074            LOG.warn("Request is missing message type - plain BOOTP not supported: " + request);
075            return null;
076        }
077
078        if (request.getHardwareAddress() == null) {
079            LOG.warn("Request is missing hardware address: " + request);
080            return null;
081        }
082
083        // From StoreBasedDhcpService
084        InetAddress serverAddress = request.getOptions().getAddressOption(ServerIdentifier.class);
085        if (!AddressUtils.isZeroAddress(serverAddress)) {
086            FIND:
087            {
088                for (InterfaceAddress localAddress : context.getInterfaceAddresses())
089                    if (Objects.equal(localAddress.getAddress(), serverAddress))
090                        break FIND;
091                LOG.debug("Request directed to " + serverAddress + ", not to this server: " + request);
092                return null;
093            }
094        }
095
096        DhcpMessage reply;
097        // dispatch based on the message type
098        switch (request.getMessageType()) {
099            case DHCPDISCOVER:
100                reply = handleDISCOVER(context, request);
101                break;
102
103            case DHCPOFFER:
104                reply = handleOFFER(context, request);
105                break;
106
107            case DHCPREQUEST:
108                reply = handleREQUEST(context, request);
109                break;
110
111            case DHCPDECLINE:
112                reply = handleDECLINE(context, request);
113                break;
114
115            case DHCPRELEASE:
116                reply = handleRELEASE(context, request);
117                break;
118
119            case DHCPINFORM:
120                reply = handleINFORM(context, request);
121                break;
122
123            case DHCPACK:
124            case DHCPNAK:
125                reply = null; // just ignore them
126                break;
127
128            default:
129                reply = handleUnknownMessage(context, request);
130                break;
131        }
132
133        return reply;
134    }

250    /**
251     * Determine address on which to base selection. If the relay agent address is
252     * set, we use the relay agent's address, otherwise we use the address we
253     * received the request from.
254     *
255     * @return InetAddress
256     */
257    @Nonnull
258    @Deprecated // This isn't used any more.
259    public static InetAddress getRemoteAddress(
260            @Nonnull InterfaceAddress localAddress,
261            @CheckForNull InetSocketAddress remoteAddress,
262            @Nonnull DhcpMessage request)
263            throws DhcpException {
264        // FIXME: do we know
265        // a) the interface address over which we received a message (!)
266        // b) the client address (if specified)
267        // c) the relay agent address?
268
269        InetAddress assignedClientAddress = request.getAssignedClientAddress();
270        if (!AddressUtils.isZeroAddress(assignedClientAddress))
271            return assignedClientAddress;
272        InetAddress currentClientAddress = request.getCurrentClientAddress();
273        if (!AddressUtils.isZeroAddress(currentClientAddress))
274            return currentClientAddress;
275        InetAddress requestedClientAddress = request.getOptions().getAddressOption(RequestedIpAddress.class);
276        if (!AddressUtils.isZeroAddress(requestedClientAddress))
277            return requestedClientAddress;
278
279        // if the relay agent address is set, we use it as the selection base
280        if (!AddressUtils.isZeroAddress(request.getRelayAgentAddress()))
281            return request.getRelayAgentAddress();
282        if (remoteAddress != null)
283            return remoteAddress.getAddress();
284        return localAddress.getAddress();
285    }

287    /**
288     * Strip options that the client doesn't want, if the ParameterRequestList
289     * option is present.
290     *
291     * @param request
292     * @param options
293     */
294    protected static void stripOptions(@Nonnull DhcpMessage request, @Nonnull OptionsField options) {
295        // these options must not be present
296        options.remove(RequestedIpAddress.class);
297        options.remove(ParameterRequestList.class);
298        options.remove(ClientIdentifier.class);
299        options.remove(MaximumDhcpMessageSize.class);
300
301        ParameterRequestList prl = request.getOptions().get(ParameterRequestList.class);
302        if (prl == null)
303            return;
304
305        byte[] tags = prl.getData();
306        Arrays.sort(tags);
307
308        for (Iterator<DhcpOption> i = options.iterator(); i.hasNext();) {
309            DhcpOption o = i.next();
310            if (o instanceof ServerIdentifier)
311                continue;
312            if (o instanceof IpAddressLeaseTime)
313                continue;
314            if (Arrays.binarySearch(tags, o.getTag()) >= 0)
315                continue;
316            i.remove();
317        }
318    }
