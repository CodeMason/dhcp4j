006package org.anarres.dhcp.common.address;
007
008import java.net.InetAddress;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011
012/**
013 * A network address: A base address and a netmask representing a contiguous
014 * range of addresses.
015 *
016 * The base address is canonicalized by the constructor to be the
017 * first address in the permissible range. Thus {@link #getAddress()}
018 * will NOT necessarily return the same {@link InetAddress} as was passed
019 * to the constructor.
020 *
021 * @see NetworkSet
022 *
023 * @author shevek
024 */
025public class NetworkAddress extends AbstractMaskedAddress {
026
027    /**
028     * Constructs a NetworkAddress from a String of the form 1.2.3.4/25.
029     *
030     * @throws IllegalArgumentException if the argument was duff.
031     */
032    @Nonnull
033    public static NetworkAddress forString(@Nonnull String addressString) {
034        return InterfaceAddress.forString(addressString).toNetworkAddress();
035    }
036
037    /**
038     * This constructor canonicalizes the fundamental address to be the network address.
039     */
040    public NetworkAddress(@Nonnull InetAddress address, @Nonnegative int netmask) {
041        super(AddressUtils.toNetworkAddress(address, netmask), netmask);
042    }
043
044    public NetworkAddress(@Nonnull java.net.InterfaceAddress address) {
045        this(address.getAddress(), address.getNetworkPrefixLength());
046    }
047
048    @Override
049    public InetAddress getNetworkAddress() {
050        return getAddress();
051    }
052
053    /**
054     * Returns the <code>index</code>th machine address in this network.
055     *
056     * The 0th address is the network address. The first usable address is index 1.
057     * If your network is larger than 2^63 machines, you are out of luck
058     * with this method.
059     *
060     * @see #getMachineInterfaceAddress(long)
061     * @see AddressUtils#add(InetAddress, long)
062     */
063    @Nonnull
064    public InetAddress getMachineAddress(@Nonnegative long index) {
065        return AddressUtils.add(getNetworkAddress(), index);
066    }
067
068    /**
069     * Returns the <code>index</code>th machine address in this network.
070     *
071     * The 0th address is the network address. The first usable address is index 1.
072     * If your network is larger than 2^63 machines, you are out of luck
073     * with this method.
074     *
075     * The netmask in the returned {@link InterfaceAddress} is that of this
076     * NetworkAddress.
077     *
078     * @see #getMachineAddress(long)
079     * @see AddressUtils#add(InetAddress, long)
080     */
081    @Nonnull
082    public InterfaceAddress getMachineInterfaceAddress(@Nonnegative long index) {
083        return new InterfaceAddress(getMachineAddress(index), getNetmask());
084    }
085
086    /**
087     * Returns true iff this network contains the given InetAddress.
088     *
089     * @see InterfaceAddress#isLocal(InetAddress)
090     * @see AddressUtils#isLocal(AbstractMaskedAddress, InetAddress)
091     */
092    public boolean contains(@Nonnull InetAddress address) {
093        return AddressUtils.isLocal(this, address);
094    }
095
096    /**
097     * Returns true iff this network contains the whole of the given network.
098     * That is to say, the given network represents a sub-network of this network.
099     */
100    public boolean contains(@Nonnull NetworkAddress networkAddress) {
101        return contains(networkAddress.getNetworkAddress())
102                && contains(networkAddress.getBroadcastAddress());
103    }
104
105    /**
106     * Returns an InetAddressRange representation of this network's address range.
107     *
108     * The InetAddressRange isn't as useful for computation, but can be
109     * useful for display or configuring other tools.
110     */
111    @Nonnull
112    public InetAddressRange toRange() {
113        return new InetAddressRange(getNetworkAddress(), getBroadcastAddress());
114    }
115}
