030    @Nonnull
031    private static byte[] extend(@Nonnull byte[] b) {
032        byte[] c = new byte[b.length + 1];
033        System.arraycopy(b, 0, c, 1, b.length);
034        return c;
035    }
036
037    @Nonnull
038    private static byte[] truncate(@Nonnull byte[] b) {
039        byte[] c = new byte[b.length - 1];
040        System.arraycopy(b, 1, c, 0, c.length);
041        return c;
042    }

044    /** Like InetAddress but without copying data in the getter. */
045    private static class Address implements Comparable<Address> {
046
047        private final byte[] data;
048
049        public Address(@Nonnull byte[] data) {
050            this.data = data;
051        }
052
053        @Nonnull
054        public byte[] getData() {
055            return data;
056        }
057
058        @Override
059        public int compareTo(@Nonnull Address o) {
060            return UnsignedBytes.lexicographicalComparator().compare(getData(), o.getData());
061        }
062
063        @Nonnull
064        public InetAddress toInetAddress() {
065            return AddressUtils.toInetAddress(truncate(data));
066        }
067
068        @Override
069        public int hashCode() {
070            return Arrays.hashCode(data);
071        }
072
073        @Override
074        public boolean equals(Object obj) {
075            if (this == obj)
076                return true;
077            if (null == obj)
078                return false;
079            if (!getClass().equals(obj.getClass()))
080                return false;
081            return Arrays.equals(data, ((Address) obj).data);
082        }
083
084        @Override
085        public String toString() {
086            return UnsignedBytes.join(" ", data);
087        }
088    }

090    private static class Count {
091
092        @CheckForSigned
093        private int value;
094
095        public Count(@CheckForSigned int value) {
096            this.value = value;
097        }
098
099        public boolean add(@CheckForSigned int delta) {
100            value += delta;
101            return value == 0;
102        }
103    }
104    private final SortedMap<Address, Count> data = new TreeMap<Address, Count>();
105
106    /**
107     * @param start The inclusive start-point.
108     * @param end The exclusive end-point.
109     */
110    private void addRange(@Nonnull byte[] start, @Nonnull byte[] end, int delta) {
111        START:
112        {
113            Address key = new Address(start);
114            Count count = data.get(key);
115            if (count == null)
116                data.put(key, new Count(delta));
117            else if (count.add(delta))
118                data.remove(key);
119        }
120
121        END:
122        {
123            Address key = new Address(end);
124            Count count = data.get(key);
125            if (count == null)
126                data.put(key, new Count(-delta));
127            else if (count.add(-delta))
128                data.remove(key);
129        }
130    }

132    /**
133     * Inclusive endpoints.
134     */
135    public void addRange(@Nonnull InetAddress start, @Nonnull InetAddress end) {
136        addRange(extend(start.getAddress()), AddressUtils.increment(extend(end.getAddress())), 1);
137    }
138
139    /**
140     * Inclusive endpoints.
141     */
142    public void removeRange(@Nonnull InetAddress start, @Nonnull InetAddress end) {
143        addRange(extend(start.getAddress()), AddressUtils.increment(extend(end.getAddress())), -1);
144    }

173    private static void _add_bit(@Nonnull byte[] data, @Nonnegative int bitIndex) {
174        // LOG.info("  +: Add " + bitIndex + " to " + UnsignedBytes.join(" ", data));
175        // LOG.info(bitIndex + " -> " + (bitIndex / 8) + "[" + (bitIndex % 8) + "] & " + byteValue);
176
177        int byteValue = 1 << (7 - (bitIndex % 8));
178        // This is actually an arbitrary precision arithmetic routine computing
179        // data + (1 e bitIndex)
180        for (int byteIndex = bitIndex / 8; byteIndex >= 0; byteIndex--) {
181            if (byteValue == 0)
182                break;
183            byteValue += UnsignedBytes.toInt(data[byteIndex]);
184            data[byteIndex] = (byte) (byteValue & 0xFF);
185            byteValue >>= Byte.SIZE;
186        }
187        // LOG.info("  +: Result is " + UnsignedBytes.join(" ", data));
188    }

190    /** endAddress is EXCLUSIVE. */
191    private static void toNetworkList(@Nonnull List<NetworkAddress> out, @Nonnull Address startAddress, @Nonnull Address endAddress) {
192        // LOG.info("(root) Reducing " + startAddress + " - " + endAddress);
193        while (startAddress.compareTo(endAddress) < 0) {
194            // LOG.info("Reducing " + startAddress + " - " + endAddress);
195            byte[] data = startAddress.getData();  // Remember, this is one byte overlength.
196            int startNetmask = Math.max(AddressUtils.toNetmask(data) - 8, 0);
197
198            byte[] tmp = new byte[data.length];
199            // Search down from the first set bit for the highest bit we can add without overflowing endAddress.
200            SEARCH:
201            {
202                for (int netmask = startNetmask; netmask < (data.length - 1) * Byte.SIZE + 1; netmask++) {
203                    System.arraycopy(data, 0, tmp, 0, data.length);
204                    // Add Byte.SIZE to correct for the overlength
205                    // Subtract 1 to hit the set bit.
206                    _add_bit(tmp, netmask + Byte.SIZE - 1);
207                    // LOG.info("   [" + netmask + "] Compare tmp=" + UnsignedBytes.join(" ", tmp) + " end=" + UnsignedBytes.join(" ", endAddress.getData()));
208                    if (UnsignedBytes.lexicographicalComparator().compare(tmp, endAddress.getData()) <= 0) {
209                        NetworkAddress networkAddress = new NetworkAddress(AddressUtils.toInetAddress(truncate(data)), netmask);
210                        // LOG.info("Adding " + networkAddress.toRange());
211                        out.add(networkAddress);
212                        break SEARCH;
213                    }
214                }
215                throw new IllegalStateException("WAT?");
216            }
217
218            startAddress = new Address(tmp);    // now we'd better make a new tmp.
219        }
220    }

222    @Nonnull
223    public List<NetworkAddress> toNetworkList() {
224        List<NetworkAddress> out = new ArrayList<NetworkAddress>();
225        Address startAddress = null;
226        int total = 0;
227        for (Map.Entry<Address, Count> e : data.entrySet()) {
228            NetworkSet.Address address = e.getKey();
229            NetworkSet.Count count = e.getValue();
230
231            if (total == 0)
232                startAddress = address;
233            total += count.value;
234            if (total == 0)
235                toNetworkList(out, startAddress, address);
236        }
237        return out;
238    }

240    @Nonnull
241    private static InetAddressRange toRange(@Nonnull Address start, @Nonnull Address end) {
242        // This routine does two copies: One here and one in truncate().
243        byte[] tmp = Arrays.copyOf(end.data, end.data.length);
244        AddressUtils.decrement(tmp);
245        return new InetAddressRange(start.toInetAddress(), AddressUtils.toInetAddress(truncate(tmp)));
246    }

248    @Nonnull
249    public List<InetAddressRange> toAddressRangeList() {
250        List<InetAddressRange> out = new ArrayList<InetAddressRange>();
251        Address startAddress = null;
252        int total = 0;
253        for (Map.Entry<Address, Count> e : data.entrySet()) {
254            NetworkSet.Address address = e.getKey();
255            NetworkSet.Count count = e.getValue();
256
257            if (total == 0)
258                startAddress = address;
259            total += count.value;
260            if (total == 0)
261                out.add(toRange(startAddress, address));
262        }
263        return out;
264    }
