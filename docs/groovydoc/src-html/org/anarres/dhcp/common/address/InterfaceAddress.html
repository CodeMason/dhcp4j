006package org.anarres.dhcp.common.address;
007
008import com.google.common.net.InetAddresses;
009import java.net.InetAddress;
010import javax.annotation.Nonnegative;
011import javax.annotation.Nonnull;
012
013/**
014 * An interface address: An address and a netmask.
015 *
016 * The address may be any address, and the lowest and highest directly
017 * accessible addresses on the network may be retrieved.
018 *
019 *
020 * @author shevek
021 */
022public class InterfaceAddress extends AbstractMaskedAddress {
023
024    /**
025     * Constructs an InterfaceAddress from a String of the form 1.2.3.4/25.
026     *
027     * @throws IllegalArgumentException if the argument was duff.
028     */
029    @Nonnull
030    public static InterfaceAddress forString(@Nonnull String addressString) {
031        String netmaskString = null;
032        int idx = addressString.indexOf('/');
033        if (idx != -1) {
034            netmaskString = addressString.substring(idx + 1);
035            addressString = addressString.substring(0, idx);
036        }
037        InetAddress address = InetAddresses.forString(addressString);
038        int netmask;
039        if (netmaskString != null) {
040            try {
041                netmask = Integer.parseInt(netmaskString);
042            } catch (NumberFormatException e) {
043                throw new IllegalArgumentException("Cannot parse netmask from " + netmaskString, e);
044            }
045        } else {
046            netmask = address.getAddress().length * Byte.SIZE;
047        }
048        return new InterfaceAddress(address, netmask);
049    }
050
051    public InterfaceAddress(@Nonnull InetAddress address, @Nonnegative int netmask) {
052        super(address, netmask);
053    }
054
055    public InterfaceAddress(@Nonnull java.net.InterfaceAddress address) {
056        this(address.getAddress(), address.getNetworkPrefixLength());
057    }
058
059    /**
060     * Returns true iff this interface address "should" be able to reach the given InetAddress.
061     *
062     * @see NetworkAddress#contains(InetAddress)
063     * @see AddressUtils#isLocal(AbstractMaskedAddress, InetAddress)
064     */
065    public boolean isLocal(@Nonnull InetAddress address) {
066        return AddressUtils.isLocal(this, address);
067    }
068
069    /**
070     * Constructs a new {@link NetworkAddress} by canonicalizing the
071     * There isn't much computational reason to do this, as all of the
072     * properties of the returned {@link NetworkAddress} will agree with this
073     * InterfaceAddress except for the fundamental {@link #getAddress() address}.
074     */
075    @Nonnull
076    public NetworkAddress toNetworkAddress() {
077        return new NetworkAddress(getAddress(), getNetmask());
078    }
079}
