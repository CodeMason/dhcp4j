031    /**
032     * Performs an arbitrary-precision increment of a byte array.
033     *
034     * @param in The array to increment.
035     * @return The same input array.
036     */
037    @Nonnull
038    public static byte[] increment(@Nonnull byte[] in) {
039        for (int i = in.length - 1; i >= 0; i--) {
040            if (UnsignedBytes.toInt(in[i]) < 255) {
041                in[i]++;
042                break;
043            }
044            in[i] = 0;
045        }
046
047        return in;
048    }

052     */
053    @Nonnull
054    public static InetAddress increment(@Nonnull InetAddress in) {
055        return toInetAddress(increment(in.getAddress()));
056    }

059     * Performs an arbitrary-precision decrement of a byte array.
060     *
061     * @param in The array to decrement.
062     * @return The same input array.
063     */
064    @Nonnull
065    public static byte[] decrement(@Nonnull byte[] in) {
066        for (int i = in.length - 1; i >= 0; i--) {
067            if (UnsignedBytes.toInt(in[i]) > 0) {
068                in[i]--;
069                break;
070            }
071            in[i] = UnsignedBytes.MAX_VALUE;
072        }
073
074        return in;
075    }

085    private static void _add(@Nonnull byte[] in, @Nonnegative long value) {
086        for (int i = in.length - 1; i >= 0; i--) {
087            if (value == 0)
088                break;
089            value += UnsignedBytes.toInt(in[i]);
090            in[i] = (byte) (value & 0xFF);
091            value >>>= Byte.SIZE;
092        }
093        // Preconditions.checkArgument(value == 0, "Carry overflow after arithmetic.");
094    }

096    /** Big-endian. Treats value as unsigned. */
097    @Nonnull
098    public static byte[] add(@Nonnull byte[] in, @Nonnegative long value) {
099        Preconditions.checkArgument(Long.SIZE - Long.numberOfLeadingZeros(value) <= in.length * Byte.SIZE,
100                "Illegal addend %s for array of length %s", value, in.length);
101        _add(in, value);
102        return in;
103    }

111    /** Big-endian. */
112    @Nonnull
113    public static byte[] add(@Nonnull byte[] in, @Nonnull byte[] value) {
114        Preconditions.checkArgument(in.length == value.length,
115                "Illegal addend of length %s for array of length %s", value.length, in.length);
116        // return new BigInteger(in).add(new BigInteger(Longs.toByteArray(value))).toByteArray();
117        int carry = 0;
118        for (int i = in.length - 1; i >= 0; i--) {
119            int sum = UnsignedBytes.toInt(in[i]) + UnsignedBytes.toInt(value[i]) + carry;
120            in[i] = (byte) (sum & 0xFF);
121            carry = sum >> Byte.SIZE;
122        }
123
124        // Preconditions.checkArgument(carry == 0, "Carry overflow after addition.");
125        return in;
126    }

131        Preconditions.checkArgument(Long.SIZE - Long.numberOfLeadingZeros(value) <= in.length * Byte.SIZE,
132                "Illegal subtrahend %s for array of length %s", value, in.length);
133        _add(in, -value);
134        return in;
135    }

142    /** Big-endian. */
143    @Nonnull
144    public static byte[] subtract(@Nonnull byte[] in, @Nonnull byte[] value) {
145        Preconditions.checkArgument(in.length == value.length,
146                "Illegal subtrahend of length %s for array of length %s", value.length, in.length);
147
148        int carry = 0;
149        for (int i = in.length - 1; i >= 0; i--) {
150            int sum = UnsignedBytes.toInt(in[i]) - UnsignedBytes.toInt(value[i]) + carry;
151            in[i] = (byte) (sum & 0xFF);
152            carry = sum >> Byte.SIZE;
153        }
154
155        // Preconditions.checkArgument(carry == 0, "Carry overflow after subtraction.");
156        return in;
157    }

159    /**
160     * Determines whether the given address is null or
161     * "0.0.0.0" (or the IPv6 equivalent).
162     */
163    public static boolean isZeroAddress(@CheckForNull byte[] address) {
164        if (address == null)
165            return true;
166        for (byte b : address)
167            if (b != 0)
168                return false;
169        return true;
170    }

203     * Determines whether the given address is a "useful" unicast address.
204     *
205     * Site local is allowed. Loopback is not.
206     */
207    public static boolean isUnicastAddress(@CheckForNull InetAddress address) {
208        return (address != null)
209                && !address.isAnyLocalAddress()
210                && !address.isLoopbackAddress()
211                && !address.isMulticastAddress();

214    /**
215     * Constructs an InetAddress from the given byte array.
216     * This is equivalent to {@link InetAddress#getByAddress(byte[])}
217     * but throws only unchecked exceptions.
218     *
219     * @throws RuntimeException if the underlying routine throws {@link UnknownHostException}.
220     */
221    @Nonnull
222    public static InetAddress toInetAddress(@Nonnull byte[] data) {
223        try {
224            return InetAddress.getByAddress(data);
225        } catch (UnknownHostException e) {
226            throw Throwables.propagate(e);
227        }
228    }

230    private static int toByteMask(@Nonnegative int netmask) {
231        // If netmask%8 is the number of bits to keep, then
232        // 8 - netmask%8 is the number of bits to remove.
233        // The & 0xff on the end is informational.
234        // Compute the number of bits which ought to be removed.
235        return (-1 << (Byte.SIZE - (netmask % Byte.SIZE))) & 0xff;
236    }

238    /**
239     * Zeros out all but the first <code>netmask</code> bits of
240     * <code>in</code>.
241     * This routine modifies its argument and returns it.
242     */
243    @Nonnull
244    public static byte[] toNetworkAddress(@Nonnull byte[] in, @Nonnegative int netmask) {
245        int idx = netmask / Byte.SIZE;
246        if (idx < in.length) {
247            int mask = toByteMask(netmask);
248            in[idx] &= mask;
249            Arrays.fill(in, idx + 1, in.length, (byte) 0);
250        }
251        return in;
252    }

255    public static InetAddress toNetworkAddress(@Nonnull InetAddress in, @Nonnegative int netmask) {
256        return toInetAddress(toNetworkAddress(in.getAddress(), netmask));
257    }

259    @Nonnull
260    public static byte[] toBroadcastAddress(@Nonnull byte[] in, @Nonnegative int netmask) {
261        int idx = netmask / Byte.SIZE;
262        if (idx < in.length) {
263            int mask = toByteMask(netmask);
264            in[idx] &= mask;
265            in[idx] |= ~mask;
266            Arrays.fill(in, idx + 1, in.length, UnsignedBytes.MAX_VALUE);
267        }
268        return in;
269    }

272    public static InetAddress toBroadcastAddress(@Nonnull InetAddress in, @Nonnegative int netmask) {
273        return toInetAddress(toBroadcastAddress(in.getAddress(), netmask));
274    }

276    /**
277     * Constructs an address suitable for use as a mask.
278     *
279     * The return value is a big-endian byte array of the specified length.
280     * The high <code>netmask</code> bits will be 1 and the remaining low bits will be 0.
281     * For example, the address may be of the form 255.255.240.0 or ff:ff:fc:....
282     */
283    @Nonnull
284    public static byte[] toNetworkMask(@Nonnegative int addressLength, @Nonnegative int netmask) {
285        byte[] out = new byte[addressLength];
286        int idx = netmask / Byte.SIZE;
287        if (idx < addressLength) {
288            Arrays.fill(out, 0, idx, UnsignedBytes.MAX_VALUE);
289            out[idx] = (byte) toByteMask(netmask);
290        } else {
291            Arrays.fill(out, UnsignedBytes.MAX_VALUE);
292        }
293        return out;
294    }

296    @Nonnull
297    public static InetAddress toNetworkMaskAddress(@Nonnegative int addressLength, @Nonnegative int netmask) {
298        return toInetAddress(toNetworkMask(addressLength, netmask));
299    }

311    /**
312     * Converts an address of the form 255.255.240.0 into an CIDR netmask.
313     */
314    @Nonnegative
315    public static int toNetmask(@Nonnull byte[] netmaskAddress) {
316        for (int i = netmaskAddress.length - 1; i >= 0; i--) {
317            // Find the last nonzero byte.
318            if (netmaskAddress[i] == 0)
319                continue;
320            // We have a nonzero byte.
321            int byteValue = UnsignedBytes.toInt(netmaskAddress[i]);
322            int ntz = Integer.numberOfTrailingZeros(byteValue);
323            return (i + 1) * Byte.SIZE - ntz;
324        }
325        return 0;
326    }

329     * Converts an address of the form 255.255.240.0 into an CIDR netmask.
330     */
331    @Nonnegative
332    public static int toNetmask(@Nonnull InetAddress netmaskAddress) {
333        return toNetmask(netmaskAddress.getAddress());
334    }

336    public static long toLong(@Nonnull byte[] data) {
337        long out = 0;
338        for (byte b : data)
339            out = (out << Byte.SIZE) + UnsignedBytes.toInt(b);
340        return out;
341    }

343    /**
344     * Returns whether a given target address is hypothetically reachable from the given AbstractMaskedAddress.
345     */
346    public static boolean isLocal(@Nonnull AbstractMaskedAddress self, @Nonnull InetAddress target) {
347        Preconditions.checkNotNull(self, "Self (AbstractMaskedAddress) was null.");
348        Preconditions.checkNotNull(target, "Target (InetAddress) was null.");
349        if (!self.getAddress().getClass().equals(target.getClass()))
350            return false;
351        byte[] selfBytes = toNetworkAddress(self.getAddress().getAddress(), self.getNetmask());
352        byte[] targetBytes = toNetworkAddress(target.getAddress(), self.getNetmask());
353        return Arrays.equals(selfBytes, targetBytes);
354    }
