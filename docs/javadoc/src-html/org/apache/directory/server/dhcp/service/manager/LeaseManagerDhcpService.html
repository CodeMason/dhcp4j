006package org.apache.directory.server.dhcp.service.manager;
007
008import java.net.InetAddress;
009import javax.annotation.Nonnull;
010import org.anarres.dhcp.common.address.AddressUtils;
011import org.apache.directory.server.dhcp.DhcpException;
012import org.apache.directory.server.dhcp.io.DhcpRequestContext;
013import org.apache.directory.server.dhcp.messages.DhcpMessage;
014import org.apache.directory.server.dhcp.messages.MessageType;
015import org.apache.directory.server.dhcp.options.dhcp.IpAddressLeaseTime;
016import org.apache.directory.server.dhcp.options.dhcp.RequestedIpAddress;
017import org.apache.directory.server.dhcp.service.AbstractDhcpService;
018import org.apache.directory.server.dhcp.service.DhcpService;
019
020/**
021 * Wraps a {@link LeaseManager} as a {@link DhcpService}.
022 *
023 * This gives slightly more convenience and automatic options management at
024 * the expense of very little loss of flexibility. This approach is generally
025 * recommended.
026 *
027 * @author shevek
028 */
029public class LeaseManagerDhcpService extends AbstractDhcpService {
030
031    @Nonnull
032    private final LeaseManager leaseManager;
033
034    public LeaseManagerDhcpService(@Nonnull LeaseManager leaseManager) {
035        this.leaseManager = leaseManager;
036    }
037
038    @Nonnull
039    public LeaseManager getLeaseManager() {
040        return leaseManager;
041    }
042
043    private static void checkReplyType(DhcpMessage request, DhcpMessage reply, MessageType... types) {
044        for (MessageType type : types)
045            if (type.equals(reply.getMessageType()))
046                return;
047        throw new IllegalStateException("Illegal response type " + reply.getMessageType() + " to request type " + request.getMessageType());
048    }
049
050    @Override
051    protected DhcpMessage handleDISCOVER(DhcpRequestContext context, DhcpMessage request) throws DhcpException {
052        // InetAddress networkAddress = getRemoteAddress(localAddresses, request, remoteAddress);
053        InetAddress requestedAddress = request.getOptions().getAddressOption(RequestedIpAddress.class);
054        long requestedExpirySecs = request.getOptions().getIntOption(IpAddressLeaseTime.class);
055        DhcpMessage reply = getLeaseManager().leaseOffer(context, request, requestedAddress, requestedExpirySecs);
056        if (reply == null)
057            return null;
058        checkReplyType(request, reply, MessageType.DHCPOFFER, MessageType.DHCPNAK);
059        setServerIdentifier(context, reply);
060        stripOptions(request, reply.getOptions());
061        return reply;
062    }
063
064    @Override
065    protected DhcpMessage handleREQUEST(DhcpRequestContext context, DhcpMessage request) throws DhcpException {
066        InetAddress requestedAddress = request.getOptions().getAddressOption(RequestedIpAddress.class);
067        if (requestedAddress == null)
068            return newReplyNak(request);
069        long requestedExpirySecs = request.getOptions().getIntOption(IpAddressLeaseTime.class);
070        DhcpMessage reply = getLeaseManager().leaseRequest(context, request, requestedAddress, requestedExpirySecs);
071        if (reply == null)
072            return newReplyNak(request);
073        checkReplyType(request, reply, MessageType.DHCPACK, MessageType.DHCPNAK);
074        setServerIdentifier(context, reply);
075        stripOptions(request, reply.getOptions());
076        return reply;
077    }
078
079    @Override
080    protected DhcpMessage handleDECLINE(DhcpRequestContext context, DhcpMessage request) throws DhcpException {
081        InetAddress declinedAddress = request.getOptions().getAddressOption(RequestedIpAddress.class);
082        if (declinedAddress == null)
083            return newReplyNak(request);
084        boolean result = getLeaseManager().leaseDecline(context, request, declinedAddress);
085        if (!result)
086            return newReplyNak(request);
087        DhcpMessage reply = newReplyAck(request, MessageType.DHCPACK, declinedAddress, -1);
088        setServerIdentifier(context, reply);
089        stripOptions(request, reply.getOptions());
090        return reply;
091    }
092
093    @Override
094    protected DhcpMessage handleRELEASE(DhcpRequestContext context, DhcpMessage request) throws DhcpException {
095        InetAddress releasedAddress = request.getOptions().getAddressOption(RequestedIpAddress.class);
096        if (AddressUtils.isZeroAddress(releasedAddress))
097            releasedAddress = context.getRemoteAddress();
098        if (AddressUtils.isZeroAddress(releasedAddress))
099            return newReplyNak(request);
100
101        boolean result = getLeaseManager().leaseRelease(context, request, releasedAddress);
102        if (!result)
103            return newReplyNak(request);
104        DhcpMessage reply = newReplyAck(request, MessageType.DHCPACK, releasedAddress, -1);
105        setServerIdentifier(context, reply);
106        stripOptions(request, reply.getOptions());
107        return reply;
108    }
109
110}
