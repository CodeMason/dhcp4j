006package org.apache.directory.server.dhcp.netty;
007
008import io.netty.bootstrap.Bootstrap;
009import io.netty.channel.Channel;
010import io.netty.channel.ChannelOption;
011import io.netty.channel.EventLoop;
012import io.netty.channel.EventLoopGroup;
013import io.netty.channel.nio.NioEventLoopGroup;
014import io.netty.channel.socket.nio.NioDatagramChannel;
015import io.netty.util.concurrent.DefaultThreadFactory;
016import java.io.IOException;
017import java.util.concurrent.ThreadFactory;
018import javax.annotation.Nonnegative;
019import javax.annotation.Nonnull;
020import javax.annotation.PostConstruct;
021import javax.annotation.PreDestroy;
022import org.apache.directory.server.dhcp.io.DhcpInterfaceManager;
023import org.apache.directory.server.dhcp.service.DhcpService;
024import org.apache.directory.server.dhcp.service.manager.LeaseManager;
025import org.apache.directory.server.dhcp.service.manager.LeaseManagerDhcpService;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029/**
030 *
031 * @author shevek
032 */
033public class DhcpServer extends DhcpInterfaceManager {
034
035    private static final Logger LOG = LoggerFactory.getLogger(DhcpServer.class);
036
037    private final DhcpService service;
038    private final int port;
039    private Channel channel;
040
041    public DhcpServer(@Nonnull DhcpService service, @Nonnegative int port) {
042        this.service = service;
043        this.port = port;
044    }
045
046    public DhcpServer(@Nonnull DhcpService service) {
047        this(service, DhcpService.SERVER_PORT);
048    }
049
050    public DhcpServer(@Nonnull LeaseManager manager, @Nonnegative int port) {
051        this(new LeaseManagerDhcpService(manager), port);
052    }
053
054    public DhcpServer(@Nonnull LeaseManager manager) {
055        this(new LeaseManagerDhcpService(manager));
056    }
057
058    @PostConstruct
059    public void start() throws IOException, InterruptedException {
060        super.start();
061
062        ThreadFactory factory = new DefaultThreadFactory("dhcp-server");
063        EventLoopGroup group = new NioEventLoopGroup(0, factory);
064
065        Bootstrap b = new Bootstrap();
066        b.group(group);
067        b.channel(NioDatagramChannel.class);
068        b.option(ChannelOption.SO_BROADCAST, true);
069        b.handler(new DhcpHandler(service, this));
070        channel = b.bind(port).sync().channel();
071    }
072
073    @PreDestroy
074    public void stop() throws IOException, InterruptedException {
075        EventLoop loop = channel.eventLoop();
076        channel.close().sync();
077        channel = null;
078        loop.shutdownGracefully();
079
080        super.stop();
081    }
082}
