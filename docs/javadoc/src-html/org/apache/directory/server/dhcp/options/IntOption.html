018package org.apache.directory.server.dhcp.options;
019
020import com.google.common.base.Preconditions;
021import com.google.common.primitives.Ints;
022import com.google.common.primitives.UnsignedInts;
023import javax.annotation.Nonnegative;
024import org.apache.directory.server.dhcp.DhcpException;
025
026/**
027 * The Dynamic Host Configuration Protocol (DHCP) provides a framework for
028 * passing configuration information to hosts on a TCP/IP network. Configuration
029 * parameters and other control information are carried in tagged data items
030 * that are stored in the 'options' field of the DHCP message. The data items
031 * themselves are also called "options."
032 * 
033 * This abstract base class is for options that carry an int value (32 bit).
034 * 
035 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
036 */
037public abstract class IntOption extends DhcpOption {
038
039    @Nonnegative
040    public long getIntValue() {
041        return UnsignedInts.toLong(Ints.fromByteArray(getData()));
042    }
043
044    public void setIntValue(@Nonnegative long value) {
045        Preconditions.checkArgument(value >> Integer.SIZE == 0, "out of range: %s", value);
046        setData(Ints.toByteArray((int) value));
047    }
048
049    @Override
050    public void validate() throws DhcpException {
051        super.validate();
052        if (getData().length != 4)
053            throw new DhcpException("Expected exactly 4 data bytes in " + this);
054    }
055
056    @Override
057    protected String toStringData() {
058        return String.valueOf(getIntValue());
059    }
060
061}
