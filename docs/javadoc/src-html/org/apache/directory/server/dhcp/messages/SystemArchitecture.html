005package org.apache.directory.server.dhcp.messages;
006
007import javax.annotation.Nonnull;
008
009/**
010 * https://tools.ietf.org/html/rfc4578
011 * @author shevek
012 */
013public enum SystemArchitecture {
014
015    Intel_x86PC(0, "Intel x86PC"),
016    NEC_PC98(1, "NEC/PC98"),
017    EFI_Itanium(2, "EFI Itanium"),
018    DEC_Alpha(3, "DEC Alpha"),
019    Arc_x86(4, "Arc x86"),
020    Intel_Lean_Client(5, "Intel Lean Client"),
021    EFI_IA32(6, "EFI IA32"),
022    EFI_BC(7, "EFI BC"),
023    EFI_Xscale(8, "EFI Xscale"),
024    EFI_x86_64(9, "EFI x86-64"),
025    UNKNOWN(-1, "Unknown");
026    private final int id;
027    private final String description;
028    /* pp */ SystemArchitecture(int id, String description) {
029        this.id = id;
030        this.description = description;
031    }
032
033    public int getId() {
034        return id;
035    }
036
037    @Nonnull
038    public String getDescription() {
039        return description;
040    }
041
042    @Override
043    public String toString() {
044        return name() + "(" + getId() + ": " + getDescription() + ")";
045    }
046
047    @Nonnull
048    public static SystemArchitecture forTypeCode(int code) {
049        for (SystemArchitecture type : SystemArchitecture.values())
050            if (type.getId() == code)
051                return type;
052        return UNKNOWN;
053    }
054}
