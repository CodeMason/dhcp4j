006package org.apache.directory.server.dhcp.service.manager;
007
008import java.net.InetAddress;
009import javax.annotation.CheckForNull;
010import javax.annotation.CheckForSigned;
011import javax.annotation.Nonnull;
012import org.apache.directory.server.dhcp.DhcpException;
013import org.apache.directory.server.dhcp.io.DhcpRequestContext;
014import org.apache.directory.server.dhcp.messages.DhcpMessage;
015
016/**
017 * A LeaseManager is easier to write than a full DhcpService.
018 * 
019 * When wrapped in a {@link LeaseManagerDhcpService}, a lot of the nasties
020 * of the DHCP protocol are taken care of.
021 *
022 * @author shevek
023 */
024public interface LeaseManager {
025
026    /**
027     * Determine a lease to offer in response to a DHCPDISCOVER message.
028     * <p>
029     * When a server receives a DHCPDISCOVER message from a client, the server
030     * chooses a network address for the requesting client. If no address is
031     * available, the server may choose to report the problem to the system
032     * administrator. If an address is available, the new address SHOULD be
033     * chosen as follows:
034     * <ul>
035     * <li> The client's current address as recorded in the client's current
036     * binding, ELSE
037     * <li> The client's previous address as recorded in the client's (now
038     * expired or released) binding, if that address is in the server's pool of
039     * available addresses and not already allocated, ELSE
040     * <li> The address requested in the 'Requested IP Address' option, if that
041     * address is valid and not already allocated, ELSE
042     * <li> A new address allocated from the server's pool of available
043     * addresses; the address is selected based on the subnet from which the
044     * message was received (if 'giaddr' is 0) or on the address of the relay
045     * agent that forwarded the message ('giaddr' when not 0).
046     * </ul>
047     */
048    @CheckForNull
049    public DhcpMessage leaseOffer(@Nonnull DhcpRequestContext context,
050            @Nonnull DhcpMessage request,
051            @CheckForNull InetAddress clientRequestedAddress, @CheckForSigned long clientRequestedExpirySecs)
052            throws DhcpException;
053
054    @CheckForNull
055    public DhcpMessage leaseRequest(@Nonnull DhcpRequestContext context,
056            @Nonnull DhcpMessage request,
057            @Nonnull InetAddress clientRequestedAddress, @CheckForSigned long clientRequestedExpirySecs)
058            throws DhcpException;
059
060    public boolean leaseDecline(@Nonnull DhcpRequestContext context,
061            @Nonnull DhcpMessage request,
062            @Nonnull InetAddress clientAddress)
063            throws DhcpException;
064
065    public boolean leaseRelease(@Nonnull DhcpRequestContext context,
066            @Nonnull DhcpMessage request,
067            @Nonnull InetAddress clientAddress)
068            throws DhcpException;
069
070}
