006package org.apache.directory.server.dhcp.messages;
007
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010
011/**
012 * https://www.iana.org/assignments/arp-parameters/arp-parameters.xhtml#arp-parameters-2
013 *
014 * @author shevek
015 */
016public enum HardwareAddressType {
017
018    Reserved(0, "Reserved"),
019    Ethernet(1, "Ethernet (10Mb)"),
020    ExperimentalEthernet(2, "Experimental Ethernet (3Mb)"),
021    AX25(3, "Amateur Radio AX.25"),
022    Proteon(4, "Proteon ProNET Token Ring"),
023    Chaos(5, "Chaos"),
024    IEEE802(6, "IEEE 802 Networks"),
025    ARCNET(7, "ARCNET"),
026    Hyperchannel(8, "Hyperchannel"),
027    Lanstar(9, "Lanstar"),
028    Autonet(10, "Autonet Short Address"),
029    LocalTalk(11, "LocalTalk"),
030    LocalNet(12, "LocalNet (IBM PCNet or SYTEK LocalNET)"),
031    Ultralink(13, "Ultra link"),
032    SMDS(14, "SMDS"),
033    FrameRelay(15, "Frame Relay"),
034    ATM_0(16, "Asynchronous Transmission Mode (ATM)"),
035    HDLC(17, "HDLC"),
036    FibreChannel(18, "Fibre Channel"),
037    ATM_1(19, "Asynchronous Transmission Mode (ATM)"),
038    SerialLine(20, "Serial Line"),
039    ATM_2(21, "Asynchronous Transmission Mode (ATM)"),
040    MIL_STD_188_220(22, "MIL-STD-188-220"),
041    Metricom(23, "Metricom"),
042    IEEE1394(24, "IEEE 1394.1995"),
043    MAPOS(25, "MAPOS"),
044    Twinaxial(26, "Twinaxial"),
045    EUI64(27, "EUI-64"),
046    HIPARP(28, "HIPARP"),
047    IPoverISO7816_3(29, "IP and ARP over ISO 7816-3"),
048    ARPSec(30, "ARPSec"),
049    IPsecTunnel(31, "IPsec tunnel"),
050    InfiniBand(32, "InfiniBand (TM)"),
051    TIA102(33, "TIA-102 Project 25 Common Air Interface (CAI)"),
052    Wiegand(34, "Wiegand Interface"),
053    PureIP(35, "Pure IP"),
054    HW_EXP1(36, "HW_EXP1"),
055    HFI(37, "HFI");
056
057    private final short code;
058    private final String description;
059
060    private HardwareAddressType(int code, String description) {
061        this.code = (short) code;
062        this.description = description;
063    }
064
065    @Nonnegative
066    public short getCode() {
067        return code;
068    }
069
070    @Nonnull
071    public String getDescription() {
072        return description;
073    }
074
075    @Override
076    public String toString() {
077        return name() + "(" + getCode() + ": " + getDescription() + ")";
078    }
079
080    @Nonnull
081    public static HardwareAddressType forCode(int code) {
082        for (HardwareAddressType type : HardwareAddressType.values())
083            if (type.code == code)
084                return type;
085        throw new IllegalArgumentException("No such hardware type " + code + ". Check "
086                + "https://www.iana.org/assignments/arp-parameters/arp-parameters.xhtml#arp-parameters-2");
087    }
088}
