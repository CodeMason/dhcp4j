020package org.apache.directory.server.dhcp.options.dhcp;
021
022import org.apache.directory.server.dhcp.options.DhcpOption;
023
024/**
025 * This option is used by DHCP clients to specify their unique
026 * identifier.  DHCP servers use this value to index their database of
027 * address bindings.  This value is expected to be unique for all
028 * clients in an administrative domain.
029 * 
030 * Identifiers SHOULD be treated as opaque objects by DHCP servers.
031 * 
032 * The client identifier MAY consist of type-value pairs similar to the
033 * 'htype'/'chaddr' fields. For instance, it MAY consist
034 * of a hardware type and hardware address. In this case the type field
035 * SHOULD be one of the ARP hardware types defined in STD2.  A
036 * hardware type of 0 (zero) should be used when the value field
037 * contains an identifier other than a hardware address (e.g. a fully
038 * qualified domain name).
039 * 
040 * For correct identification of clients, each client's client-
041 * identifier MUST be unique among the client-identifiers used on the
042 * subnet to which the client is attached.  Vendors and system
043 * administrators are responsible for choosing client-identifiers that
044 * meet this requirement for uniqueness.
045 * 
046 * The code for this option is 61, and its minimum length is 2.
047 * 
048 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
049 */
050public class ClientIdentifier extends DhcpOption {
051    /*
052     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
053     */
054
055    @Override
056    public byte getTag() {
057        return 61;
058    }
059}
