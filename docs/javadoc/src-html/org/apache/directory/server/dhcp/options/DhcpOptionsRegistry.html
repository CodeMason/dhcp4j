226    @Nonnull
227    public static DhcpOptionsRegistry getInstance() {
228        return Inner.INSTANCE;
229    }
230    private final Map<Byte, Class<? extends DhcpOption>> optionTypes = new HashMap<Byte, Class<? extends DhcpOption>>();
231    private final Map<Class<? extends DhcpOption>, Byte> optionTags = new WeakHashMap<Class<? extends DhcpOption>, Byte>();
232
233    @Nonnull
234    public static <T extends DhcpOption> T newInstance(@Nonnull Class<T> type) {
235        try {
236            return type.newInstance();
237        } catch (InstantiationException e) {
238            throw new IllegalArgumentException("Cannot instantiate " + type, e);
239        } catch (IllegalAccessException e) {
240            throw new IllegalArgumentException("Cannot instantiate " + type, e);
241        }
242    }
243
244    private byte getTagFrom(@Nonnull Class<? extends DhcpOption> type) {
245        DhcpOption o = newInstance(type);
246        byte tag = o.getTag();
247        optionTags.put(type, tag);
248        return tag;
249    }
250
251    public void addOptionType(@Nonnull Class<? extends DhcpOption> type) {
252        byte tag = getTagFrom(type);
253        if (optionTypes.put(tag, type) != null)
254            throw new IllegalArgumentException("Duplicate tag: " + type);
255        optionTags.put(type, tag);
256    }
257
258    @CheckForNull
259    public Class<? extends DhcpOption> getOptionType(byte tag) {
260        return optionTypes.get(tag);
261    }
262
263    @Nonnull
264    public byte getOptionTag(@Nonnull Class<? extends DhcpOption> type) {
265        Byte tag = optionTags.get(type);
266        if (tag != null)
267            return tag.byteValue();
268        // TODO: Warn about unregistered option.
269        return getTagFrom(type);
270    }
271
272    @Override
273    public String toString() {
274        return getClass().getSimpleName() + "(" + optionTypes + ")";
275    }
