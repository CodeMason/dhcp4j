006package org.apache.directory.server.dhcp.netty;
007
008import io.netty.bootstrap.Bootstrap;
009import io.netty.channel.EventLoop;
010import io.netty.channel.EventLoopGroup;
011import io.netty.channel.nio.NioEventLoopGroup;
012import io.netty.channel.socket.nio.NioDatagramChannel;
013import io.netty.util.concurrent.DefaultThreadFactory;
014import java.io.IOException;
015import java.net.InetAddress;
016import java.net.InetSocketAddress;
017import java.net.NetworkInterface;
018import java.util.Enumeration;
019import java.util.concurrent.ThreadFactory;
020import javax.annotation.Nonnegative;
021import javax.annotation.Nonnull;
022import javax.annotation.PostConstruct;
023import javax.annotation.PreDestroy;
024import org.anarres.dhcp.v6.options.DuidOption;
025import org.anarres.dhcp.v6.service.Dhcp6LeaseManager;
026import org.anarres.dhcp.v6.service.Dhcp6Service;
027import org.anarres.dhcp.v6.service.LeaseManagerDhcp6Service;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031/**
032 *
033 * @author marosmars
034 */
035public class Dhcp6Server {
036
037    private static final Logger LOG = LoggerFactory.getLogger(Dhcp6Server.class);
038
039    private final Dhcp6Service service;
040    private final int port;
041    private NioDatagramChannel channel;
042    private static final byte[] SERVER_ID = new byte[] { 0, 1 }; // TODO invalid type code
043    // see https://tools.ietf.org/html/rfc3315#section-9.2
044
045    public Dhcp6Server(@Nonnull Dhcp6Service service, @Nonnegative int port) {
046        this.service = service;
047        this.port = port;
048    }
049
050    public Dhcp6Server(@Nonnull Dhcp6Service service) {
051        this(service, Dhcp6Service.SERVER_PORT);
052    }
053
054    public Dhcp6Server(@Nonnull Dhcp6LeaseManager manager, @Nonnegative int port, final DuidOption.Duid serverId) {
055        this(new LeaseManagerDhcp6Service(manager, serverId), port);
056    }
057
058    public Dhcp6Server(@Nonnull Dhcp6LeaseManager manager, final DuidOption.Duid serverId) {
059        this(new LeaseManagerDhcp6Service(manager, serverId));
060    }
061
062    @PostConstruct
063    public void start() throws IOException, InterruptedException {
064        ThreadFactory factory = new DefaultThreadFactory("dhcp-server");
065        EventLoopGroup group = new NioEventLoopGroup(0, factory);
066
067        Bootstrap b = new Bootstrap();
068        b.group(group);
069        b.channel(NioDatagramChannel.class);
070        b.handler(new Dhcp6Handler(service, SERVER_ID));
071        channel = (NioDatagramChannel) b.bind(port).sync().channel();
072
073        LOG.info("DHCPv6 server started on : {}, with id: {}", channel.localAddress(), SERVER_ID);
074
075        final Enumeration<NetworkInterface> networkInterfaces = NetworkInterface.getNetworkInterfaces();
076        while (networkInterfaces.hasMoreElements()) {
077            final NetworkInterface netIf = networkInterfaces.nextElement();
078
079            final InetSocketAddress relayGroup = new InetSocketAddress(InetAddress.getByName("FF02::1:2"), port);
080            channel.joinGroup(relayGroup, netIf);
081            final InetSocketAddress serverGroup = new InetSocketAddress(InetAddress.getByName("FF02::1:3"), port);
082            channel.joinGroup(serverGroup, netIf);
083        }
084    }
085
086    @PreDestroy
087    public void stop() throws IOException, InterruptedException {
088        EventLoop loop = channel.eventLoop();
089        channel.close().sync();
090        channel = null;
091        loop.shutdownGracefully();
092    }
093}
