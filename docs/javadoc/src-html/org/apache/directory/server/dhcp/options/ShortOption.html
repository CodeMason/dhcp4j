020package org.apache.directory.server.dhcp.options;
021
022import com.google.common.base.Preconditions;
023import com.google.common.primitives.Shorts;
024import javax.annotation.Nonnegative;
025import org.apache.directory.server.dhcp.DhcpException;
026
027/**
028 * The Dynamic Host Configuration Protocol (DHCP) provides a framework for
029 * passing configuration information to hosts on a TCP/IP network. Configuration
030 * parameters and other control information are carried in tagged data items
031 * that are stored in the 'options' field of the DHCP message. The data items
032 * themselves are also called "options."
033 * 
034 * This abstract base class is for options that carry an unsigned short value
035 * (16 bit).
036 * 
037 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
038 */
039public abstract class ShortOption extends DhcpOption {
040
041    @Nonnegative
042    public int getShortValue() {
043        return Shorts.fromByteArray(getData()) & 0xFFFF;
044    }
045
046    public void setShortValue(@Nonnegative int value) {
047        Preconditions.checkArgument(value >> Short.SIZE == 0, "out of range: %s", value);
048        setData(Shorts.toByteArray((short) value));
049    }
050
051    @Override
052    public void validate() throws DhcpException {
053        super.validate();
054        if (getData().length != 2)
055            throw new DhcpException("Expected exactly 2 data bytes in " + this);
056    }
057
058    @Override
059    protected String toStringData() {
060        return String.valueOf(getShortValue());
061    }
062
063}
