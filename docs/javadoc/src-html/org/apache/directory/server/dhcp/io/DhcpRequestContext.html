006package org.apache.directory.server.dhcp.io;
007
008import com.google.common.base.Preconditions;
009import java.net.InetAddress;
010import java.net.InetSocketAddress;
011import java.util.Arrays;
012import java.util.Collections;
013import javax.annotation.CheckForNull;
014import javax.annotation.Nonnull;
015import javax.annotation.concurrent.NotThreadSafe;
016import org.anarres.dhcp.common.address.AddressUtils;
017import org.anarres.dhcp.common.address.InterfaceAddress;
018
019/**
020 * This object is mutable and should not be reused.
021 *
022 * @author shevek
023 */
024@NotThreadSafe
025public class DhcpRequestContext {
026
027    private final Iterable<? extends InterfaceAddress> interfaceAddresses;
028    private final InetSocketAddress remoteSocketAddress;
029    private final InetSocketAddress localSocketAddress;
030    private InterfaceAddress interfaceAddress = null;
031    private InetAddress clientAddress = null;
032
033    public DhcpRequestContext(
034            @Nonnull Iterable<? extends InterfaceAddress> interfaceAddresses,
035            @Nonnull InetSocketAddress remoteSocketAddress,
036            @Nonnull InetSocketAddress localSocketAddress
037    ) {
038        this.interfaceAddresses = interfaceAddresses;
039        this.remoteSocketAddress = remoteSocketAddress;
040        this.localSocketAddress = localSocketAddress;
041
042        for (InterfaceAddress interfaceAddress : interfaceAddresses)
043            if (AddressUtils.isZeroAddress(interfaceAddress.getAddress()))
044                throw new IllegalArgumentException("Illegal InterfaceAddress " + interfaceAddress);
045    }
046
047    public DhcpRequestContext(
048            @Nonnull InterfaceAddress[] interfaceAddresses,
049            @Nonnull InetSocketAddress remoteSocketAddress,
050            @Nonnull InetSocketAddress localSocketAddress) {
051        this(Arrays.asList(interfaceAddresses), remoteSocketAddress, localSocketAddress);
052    }
053
054    public DhcpRequestContext(
055            @Nonnull InterfaceAddress interfaceAddress,
056            @Nonnull InetSocketAddress remoteSocketAddress,
057            @Nonnull InetSocketAddress localSocketAddress) {
058        this(Collections.singletonList(interfaceAddress), remoteSocketAddress, localSocketAddress);
059        this.interfaceAddress = interfaceAddress;
060    }
061
062    @Nonnull
063    public Iterable<? extends InterfaceAddress> getInterfaceAddresses() {
064        return interfaceAddresses;
065    }
066
067    /** May be the zero address. */
068    @Nonnull
069    public InetSocketAddress getRemoteSocketAddress() {
070        return remoteSocketAddress;
071    }
072
073    /** May be the zero address. */
074    @Nonnull
075    public InetAddress getRemoteAddress() {
076        return Preconditions.checkNotNull(getRemoteSocketAddress().getAddress(), "Remote InetSocketAddress was null.");
077    }
078
079    /** May be the zero address. */
080    @Nonnull
081    public InetSocketAddress getLocalSocketAddress() {
082        return localSocketAddress;
083    }
084
085    /** May be the zero address. */
086    @Nonnull
087    public InetAddress getLocalAddress() {
088        return Preconditions.checkNotNull(getLocalSocketAddress().getAddress(), "Local InetAddress was null.");
089    }
090
091    public void setRelayAgentAddress(@CheckForNull InetAddress relayAgentAddress) {
092    }
093
094    @Nonnull
095    public InetAddress getClientAddress() {
096        return clientAddress;
097    }
098
099    public void setClientAddress(@Nonnull InetAddress clientAddress) {
100        Preconditions.checkArgument(!AddressUtils.isZeroAddress(clientAddress), "Client address was null or zero.");
101        if (clientAddress.equals(this.clientAddress))
102            return;
103        Preconditions.checkState(this.clientAddress == null, "Already have a (different) client address.");
104        this.clientAddress = clientAddress;
105        setInterfaceAddressFrom(clientAddress);
106    }
107
108    /**
109     * Returns the single InterfaceAddress on which this request will be transmitted.
110     *
111     * You may have to call {@link #setClientAddress(java.net.InetAddress)} before this is valid.
112     */
113    @Nonnull
114    public InterfaceAddress getInterfaceAddress() {
115        return Preconditions.checkNotNull(interfaceAddress, "InterfaceAddress not set - did you call setClientAddress()?");
116    }
117
118    private void setInterfaceAddressFrom(@Nonnull InetAddress address) {
119        IFACE:
120        if (interfaceAddress == null) {
121            for (InterfaceAddress interfaceAddress : getInterfaceAddresses()) {
122                if (interfaceAddress.isLocal(clientAddress)) {
123                    this.interfaceAddress = interfaceAddress;
124                    break IFACE;
125                }
126            }
127            throw new IllegalArgumentException("No interface can reach client address " + clientAddress + ": " + getInterfaceAddresses());
128        }
129    }
130
131}
