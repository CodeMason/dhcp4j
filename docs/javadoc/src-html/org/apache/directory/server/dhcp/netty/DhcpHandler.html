006package org.apache.directory.server.dhcp.netty;
007
008import io.netty.buffer.ByteBuf;
009import io.netty.channel.ChannelHandler;
010import io.netty.channel.ChannelHandlerContext;
011import io.netty.channel.SimpleChannelInboundHandler;
012import io.netty.channel.socket.DatagramPacket;
013import java.net.InetSocketAddress;
014import java.net.SocketAddress;
015import java.nio.ByteBuffer;
016import javax.annotation.Nonnull;
017import org.anarres.dhcp.common.MDCUtils;
018import org.anarres.dhcp.common.address.InterfaceAddress;
019import org.apache.directory.server.dhcp.io.DhcpInterfaceManager;
020import org.apache.directory.server.dhcp.io.DhcpInterfaceUtils;
021import org.apache.directory.server.dhcp.io.DhcpMessageDecoder;
022import org.apache.directory.server.dhcp.io.DhcpMessageEncoder;
023import org.apache.directory.server.dhcp.io.DhcpRequestContext;
024import org.apache.directory.server.dhcp.messages.DhcpMessage;
025import org.apache.directory.server.dhcp.service.DhcpService;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.slf4j.MDC;
029
030/**
031 *
032 * @author shevek
033 */
034@ChannelHandler.Sharable
035public class DhcpHandler extends SimpleChannelInboundHandler<DatagramPacket> {
036
037    private static final Logger LOG = LoggerFactory.getLogger(DhcpHandler.class);
038    private final DhcpService dhcpService;
039    private final DhcpInterfaceManager interfaceManager;
040    private final DhcpMessageDecoder decoder = new DhcpMessageDecoder();
041    private final DhcpMessageEncoder encoder = new DhcpMessageEncoder();
042
043    public DhcpHandler(@Nonnull DhcpService dhcpService, @Nonnull DhcpInterfaceManager interfaceManager) {
044        this.dhcpService = dhcpService;
045        this.interfaceManager = interfaceManager;
046    }
047
048    private static void debug(@Nonnull String event, @Nonnull SocketAddress src, @Nonnull SocketAddress dst, @Nonnull DhcpMessage msg) {
049        if (LOG.isDebugEnabled())
050            LOG.debug("{} {} -> {} {}", event, src, dst, msg);
051    }
052
053    @Override
054    protected void channelRead0(ChannelHandlerContext ctx, DatagramPacket msg) throws Exception {
055        DhcpMessage request = decoder.decode(msg.content().nioBuffer());
056
057        DhcpRequestContext context = interfaceManager.newRequestContext(
058                (InetSocketAddress) ctx.channel().localAddress(),
059                msg.sender(),
060                msg.recipient(),
061                request
062        );
063        if (context == null) {
064            debug("IGNQUERY", msg.sender(), msg.recipient(), request);
065            return;
066        }
067        // debug("READ", msg.sender(), msg.recipient(), request);
068
069        MDCUtils.init(context, request);
070        try {
071            DhcpMessage reply = dhcpService.getReplyFor(context, request);
072            if (reply == null) {
073                debug("NOREPLY", msg.sender(), msg.recipient(), request);
074                return;
075            }
076
077            InterfaceAddress localAddress = interfaceManager.getResponseInterface(
078                    request.getRelayAgentAddress(),
079                    request.getCurrentClientAddress(),
080                    msg.sender().getAddress(),
081                    reply
082            );
083            if (localAddress == null) {
084                debug("NOIFACE", msg.recipient(), msg.sender(), reply);
085                return;
086            }
087
088            debug("READ", msg.sender(), msg.recipient(), request);
089
090            InetSocketAddress isa = DhcpInterfaceUtils.determineMessageDestination(
091                    request, reply,
092                    localAddress, msg.sender().getPort());
093
094            ByteBuf buf = ctx.alloc().buffer(1024);
095            ByteBuffer buffer = buf.nioBuffer(buf.writerIndex(), buf.writableBytes());
096            encoder.encode(buffer, reply);
097            buffer.flip();
098            buf.writerIndex(buf.writerIndex() + buffer.remaining());
099            DatagramPacket packet = new DatagramPacket(buf, isa);
100            debug("WRITE", packet.sender(), packet.recipient(), reply);
101            ctx.write(packet, ctx.voidPromise());
102        } finally {
103            MDCUtils.fini();
104        }
105    }
106
107    @Override
108    public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
109        ctx.flush();
110    }
111
112    @Override
113    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
114        LOG.error("Error on channel: " + cause, cause);
115        // ctx.close();
116    }
117}
