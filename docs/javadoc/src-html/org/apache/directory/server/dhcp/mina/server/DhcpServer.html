006package org.apache.directory.server.dhcp.mina.server;
007
008import java.io.IOException;
009import java.net.InetSocketAddress;
010import java.net.SocketException;
011import javax.annotation.Nonnegative;
012import javax.annotation.Nonnull;
013import javax.annotation.PostConstruct;
014import javax.annotation.PreDestroy;
015import org.apache.directory.server.dhcp.io.DhcpInterfaceManager;
016import org.apache.directory.server.dhcp.mina.protocol.DhcpProtocolCodecFactory;
017import org.apache.directory.server.dhcp.mina.protocol.DhcpProtocolHandler;
018import org.apache.directory.server.dhcp.service.DhcpService;
019import org.apache.directory.server.dhcp.service.manager.LeaseManager;
020import org.apache.directory.server.dhcp.service.manager.LeaseManagerDhcpService;
021import org.apache.mina.filter.codec.ProtocolCodecFilter;
022import org.apache.mina.filter.logging.LoggingFilter;
023import org.apache.mina.transport.socket.nio.NioDatagramAcceptor;
024
025/**
026 *
027 * @author shevek
028 */
029public class DhcpServer extends DhcpInterfaceManager {
030
031    private final LoggingFilter logger_wire = new LoggingFilter("dhcp-wire");
032    private final ProtocolCodecFilter codec = new ProtocolCodecFilter(DhcpProtocolCodecFactory.getInstance());
033    private final LoggingFilter logger_packet = new LoggingFilter("dhcp-packet");
034    private final DhcpService service;
035    private final int port;
036    private NioDatagramAcceptor acceptor;
037
038    public DhcpServer(@Nonnull DhcpService service, @Nonnegative int port) {
039        this.service = service;
040        this.port = port;
041    }
042
043    public DhcpServer(@Nonnull DhcpService service) {
044        this(service, DhcpService.SERVER_PORT);
045    }
046
047    public DhcpServer(@Nonnull LeaseManager leaseManager) throws SocketException {
048        this(new LeaseManagerDhcpService(leaseManager));
049    }
050
051    @PostConstruct
052    @Override
053    public void start() throws IOException, InterruptedException {
054        super.start();
055
056        NioDatagramAcceptor a = new NioDatagramAcceptor();
057        a.bind(new InetSocketAddress(port));
058        a.getFilterChain().addLast("dhcp-wire", logger_wire);
059        a.getFilterChain().addLast("dhcp-codec", codec);
060        a.getFilterChain().addLast("dhcp-packet", logger_packet);
061        a.setHandler(new DhcpProtocolHandler(service, this));
062        this.acceptor = a;
063    }
064
065    // @NotThreadSafe
066    @PreDestroy
067    @Override
068    public void stop() throws IOException, InterruptedException {
069        acceptor.unbind();
070        acceptor = null;
071
072        super.stop();
073    }
074}
