020package org.apache.directory.server.dhcp.options.misc;
021
022import org.apache.directory.server.dhcp.options.DhcpOption;
023
024/**
025 * This option is used by clients and servers to exchange vendor-
026 * specific information.  The information is an opaque object of n
027 * octets, presumably interpreted by vendor-specific code on the clients
028 * and servers.  The definition of this information is vendor specific.
029 * The vendor is indicated in the vendor class identifier option.
030 * Servers not equipped to interpret the vendor-specific information
031 * sent by a client MUST ignore it (although it may be reported).
032 * Clients which do not receive desired vendor-specific information
033 * SHOULD make an attempt to operate without it, although they may do so
034 * (and announce they are doing so) in a degraded mode.
035 * 
036 * If a vendor potentially encodes more than one item of information in
037 * this option, then the vendor SHOULD encode the option using
038 * "Encapsulated vendor-specific options" as described below:
039 * The Encapsulated vendor-specific options field SHOULD be encoded as a
040 * sequence of code/length/value fields of identical syntax to the DHCP
041 * options field with the following exceptions:
042 * 
043 *    1) There SHOULD NOT be a "magic cookie" field in the encapsulated
044 *       vendor-specific extensions field.
045 * 
046 *    2) Codes other than 0 or 255 MAY be redefined by the vendor within
047 *       the encapsulated vendor-specific extensions field, but SHOULD
048 *       conform to the tag-length-value syntax defined in section 2.
049 * 
050 *    3) Code 255 (END), if present, signifies the end of the
051 *       encapsulated vendor extensions, not the end of the vendor
052 *       extensions field. If no code 255 is present, then the end of
053 *       the enclosing vendor-specific information field is taken as the
054 *       end of the encapsulated vendor-specific extensions field.
055 * 
056 * The code for this option is 43 and its minimum length is 1.
057 * 
058 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
059 */
060public class VendorSpecificInformation extends DhcpOption {
061    /*
062     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
063     */
064
065    @Override
066    public byte getTag() {
067        return 43;
068    }
069}
