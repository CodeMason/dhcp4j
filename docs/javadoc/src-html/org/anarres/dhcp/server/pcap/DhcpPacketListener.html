006package org.anarres.dhcp.server.pcap;

008import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
009import java.net.InetSocketAddress;
010import java.nio.ByteBuffer;
011import java.util.Arrays;
012import javax.annotation.Nonnull;
013import org.anarres.dhcp.common.address.InterfaceAddress;
014import org.apache.directory.server.dhcp.io.DhcpMessageDecoder;
015import org.apache.directory.server.dhcp.io.DhcpMessageEncoder;
016import org.apache.directory.server.dhcp.io.DhcpRequestContext;
017import org.apache.directory.server.dhcp.messages.DhcpMessage;
018import org.apache.directory.server.dhcp.service.DhcpService;
019import org.pcap4j.core.PacketListener;
020import org.pcap4j.packet.EthernetPacket;
021import org.pcap4j.packet.IpV4Packet;
022import org.pcap4j.packet.Packet;
023import org.pcap4j.packet.UdpPacket;
024import org.pcap4j.packet.UnknownPacket;
025import org.pcap4j.packet.namednumber.EtherType;
026import org.pcap4j.packet.namednumber.IpNumber;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 *
032 * @author shevek
033 */
034public class DhcpPacketListener implements PacketListener {
035
036    private static final Logger LOG = LoggerFactory.getLogger(DhcpPacketListener.class);
037    private final DhcpMessageDecoder decoder = new DhcpMessageDecoder();
038    private final DhcpMessageEncoder encoder = new DhcpMessageEncoder();
039    private final DhcpService service;
040    private final InterfaceAddress[] interfaceAddresses;
041
042    @SuppressFBWarnings("EI_EXPOSE_REP2")
043    public DhcpPacketListener(@Nonnull DhcpService service, @Nonnull InterfaceAddress[] interfaceAddresses) {
044        this.service = service;
045        this.interfaceAddresses = interfaceAddresses;
046    }
047
048    @Override
049    public void gotPacket(Packet rawPacket) {
050        try {
051            LOG.info("Read raw " + rawPacket);
052            IpV4Packet ipPacket = rawPacket.get(IpV4Packet.class);
053            UdpPacket udpPacket = rawPacket.get(UdpPacket.class);
054            byte[] dhcpData = udpPacket.getPayload().getRawData();
055            InetSocketAddress remoteAddress = new InetSocketAddress(ipPacket.getHeader().getSrcAddr(), udpPacket.getHeader().getSrcPort().valueAsInt());
056            InetSocketAddress localAddress = new InetSocketAddress(ipPacket.getHeader().getDstAddr(), udpPacket.getHeader().getDstPort().valueAsInt());
057            DhcpRequestContext context = new DhcpRequestContext(interfaceAddresses, remoteAddress, localAddress);
058            DhcpMessage request = decoder.decode(ByteBuffer.wrap(dhcpData));
059            LOG.info("Read DHCP " + request);
060            DhcpMessage reply = service.getReplyFor(context, request);
061            if (reply == null)
062                return;
063            byte[] replyData = new byte[1536];
064            ByteBuffer buffer = ByteBuffer.wrap(replyData);
065            encoder.encode(buffer, reply);
066            replyData = Arrays.copyOf(replyData, buffer.position());    // Truncate array to writer position.
067            UnknownPacket.Builder dhcpBuilder = new UnknownPacket.Builder()
068                    .rawData(replyData);
069            UdpPacket.Builder udpBuilder = new UdpPacket.Builder()
070                    .payloadBuilder(dhcpBuilder)
071                    .srcPort(udpPacket.getHeader().getDstPort())
072                    .dstPort(udpPacket.getHeader().getSrcPort())
073                    .correctChecksumAtBuild(true)
074                    .correctLengthAtBuild(true);
075            IpV4Packet.Builder ipBuilder = new IpV4Packet.Builder()
076                    .payloadBuilder(udpBuilder)
077                    .srcAddr(null) // TODO
078                    .dstAddr(null) // TODO
079                    .protocol(IpNumber.UDP)
080                    .correctChecksumAtBuild(true)
081                    .correctLengthAtBuild(true)
082                    .paddingAtBuild(true);
083            EthernetPacket.Builder ethernetBuilder = new EthernetPacket.Builder()
084                    .payloadBuilder(ipBuilder)
085                    .type(EtherType.IPV4)
086                    .paddingAtBuild(true);
087            Packet replyPacket = ethernetBuilder.build();
088            // handle.sendPacket(replyPacket);
089        } catch (Exception e) {
090            LOG.error("DHCP failed", e);
091        }
092    }
093
094}
