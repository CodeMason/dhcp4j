005package org.anarres.dhcp.common;
006
007import javax.annotation.Nonnull;
008import org.apache.directory.server.dhcp.io.DhcpRequestContext;
009import org.apache.directory.server.dhcp.messages.DhcpMessage;
010import org.slf4j.MDC;
011
012/**
013 *
014 * @author shevek
015 */
016public class MDCUtils {
017
018    public static final String MDC_DHCP_MESSAGE_TYPE = "dhcp.messageType";
019    public static final String MDC_DHCP_CLIENT_HARDWARE_ADDRESS = "dhcp.clientHardwareAddress";
020    public static final String MDC_DHCP_SERVER_INTERFACE_ADDRESS = "dhcp.serverInterfaceAddress";
021
022    public static void init(@Nonnull DhcpRequestContext context, @Nonnull DhcpMessage request) {
023        MDC.put(MDC_DHCP_MESSAGE_TYPE, String.valueOf(request.getMessageType()));
024        MDC.put(MDC_DHCP_CLIENT_HARDWARE_ADDRESS, String.valueOf(request.getHardwareAddress()));
025        MDC.put(MDC_DHCP_SERVER_INTERFACE_ADDRESS, String.valueOf(context));
026    }
027
028    public static void fini() {
029        MDC.remove(MDC_DHCP_SERVER_INTERFACE_ADDRESS);
030        MDC.remove(MDC_DHCP_CLIENT_HARDWARE_ADDRESS);
031        MDC.remove(MDC_DHCP_MESSAGE_TYPE);
032    }
033
034    private MDCUtils() {
035    }
036}
