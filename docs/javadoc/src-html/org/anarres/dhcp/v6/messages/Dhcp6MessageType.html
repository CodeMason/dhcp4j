005package org.anarres.dhcp.v6.messages;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010
011/**
012 * https://www.iana.org/assignments/dhcpv6-parameters/dhcpv6-parameters.xml
013 *
014 * @author shevek
015 */
016public enum Dhcp6MessageType {
017
018    DHCP_UNRECOGNIZED(0),
019    /** RFC3315 */
020    DHCP_SOLICIT(1),
021    /** RFC3315 */
022    DHCP_ADVERTISE(2),
023    /** RFC3315 */
024    DHCP_REQUEST(3),
025    /** RFC3315 */
026    DHCP_CONFIRM(4),
027    /** RFC3315 */
028    DHCP_RENEW(5),
029    /** RFC3315 */
030    DHCP_REBIND(6),
031    /** RFC3315 */
032    DHCP_REPLY(7),
033    /** RFC3315 */
034    DHCP_RELEASE(8),
035    /** RFC3315 */
036    DHCP_DECLINE(9),
037    /** RFC3315 */
038    DHCP_RECONFIGURE(10),
039    /** RFC3315 */
040    DHCP_INFORMATION_REQUEST(11),
041    /** RFC3315 */
042    DHCP_RELAY_FORW(12),
043    /** RFC3315 */
044    DHCP_RELAY_REPL(13),
045    /** RFC5007 */
046    LEASEQUERY(14),
047    /** RFC5007 */
048    LEASEQUERY_REPLY(15),
049    /** RFC5460 */
050    LEASEQUERY_DONE(16),
051    /** RFC5460 */
052    LEASEQUERY_DATA(17),
053    /** RFC6977 */
054    RECONFIGURE_REQUEST(18),
055    /** RFC6977 */
056    RECONFIGURE_REPLY(19),
057    /** RFC7341 */
058    DHCPV4_QUERY(20),
059    /** RFC7341 */
060    DHCPV4_RESPONSE(21);
061    private final byte code;
062
063    private Dhcp6MessageType(@Nonnegative int code) {
064        this.code = UnsignedBytes.checkedCast(code);
065    }
066
067    public byte getCode() {
068        return code;
069    }
070
071    @Nonnull
072    public static Dhcp6MessageType forTypeCode(byte type) {
073        for (Dhcp6MessageType mt : Dhcp6MessageType.values()) {
074            if (type == mt.getCode()) {
075                return mt;
076            }
077        }
078        return DHCP_UNRECOGNIZED;
079    }
080}
