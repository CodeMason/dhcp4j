005package org.anarres.dhcp.v6.io;
006
007import com.google.common.io.BaseEncoding;
008import com.google.common.primitives.Ints;
009import java.io.IOException;
010import java.net.InetAddress;
011import java.net.UnknownHostException;
012import java.nio.ByteBuffer;
013import javax.annotation.Nonnegative;
014import javax.annotation.Nonnull;
015import org.anarres.dhcp.v6.Dhcp6Exception;
016import org.anarres.dhcp.v6.messages.Dhcp6Message;
017import org.anarres.dhcp.v6.messages.Dhcp6MessageType;
018import org.anarres.dhcp.v6.messages.Dhcp6RelayMessage;
019import org.anarres.dhcp.v6.options.Dhcp6Option;
020import org.anarres.dhcp.v6.options.Dhcp6Options;
021import org.anarres.dhcp.v6.options.Dhcp6OptionsRegistry;
022import org.anarres.dhcp.v6.options.UnrecognizedOption;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 *
028 * @author shevek, marosmars
029 */
030public class Dhcp6MessageDecoder {
031
032    private static final Logger LOG = LoggerFactory.getLogger(Dhcp6MessageDecoder.class);
033
034    private static class Inner {
035        private static final Dhcp6MessageDecoder INSTANCE = new Dhcp6MessageDecoder(Dhcp6OptionsRegistry.getInstance());
036    }
037
038    @Nonnull
039    public static Dhcp6MessageDecoder getInstance() {
040        return Inner.INSTANCE;
041    }
042
043    private final Dhcp6OptionsRegistry registry;
044
045    public Dhcp6MessageDecoder(final Dhcp6OptionsRegistry registry) {
046        this.registry = registry;
047    }
048
049    // TODO make this extensible, right now we fail for every undefined DHCP message
050    // TODO extract option decoder and use just the decoder for VendorSpecificInformationOption like options
051
052    /**
053     * https://tools.ietf.org/html/rfc3315#section-6
054     * https://tools.ietf.org/html/rfc3315#section-7
055     */
056    public Dhcp6Message decode(@Nonnull ByteBuffer buffer) throws Dhcp6Exception, IOException {
057        LOG.debug("Decoding Dhcp6 message: {}", buffer);
058
059        final Dhcp6MessageType msgType = getMsgType(buffer);
060        LOG.debug("Message type: {}", msgType);
061
062        Dhcp6Message message;
063        if(isRelayedMessage(msgType)) {
064            LOG.debug("Message relay");
065
066            final Dhcp6RelayMessage relayMsg = new Dhcp6RelayMessage();
067            message = relayMsg;
068
069            relayMsg.setHopCount(getByte(buffer));
070            LOG.debug("Hop count: {}", message.getTransactionId());
071
072            relayMsg.setLinkAddress(getInetAddress(buffer, (byte) 16));
073            LOG.debug("Link address: {}", message.getTransactionId());
074
075            relayMsg.setPeerAddress(getInetAddress(buffer, (byte) 16));
076            LOG.debug("Peer address: {}", message.getTransactionId());
077        } else {
078            message = new Dhcp6Message();
079            message.setMessageType(msgType);
080
081            message.setTransactionId(getTxId(buffer));
082            LOG.debug("Transaction ID: {}", message.getTransactionId());
083
084            Dhcp6Options options = decodeOptions(buffer);
085            message.setOptions(options);
086        }
087
088        LOG.debug("Dhcp6 message decoded: {}", message);
089        message.setMessageType(msgType);
090        return message;
091
092    }
093
094    private InetAddress getInetAddress(final ByteBuffer bufferm, final byte size) throws Dhcp6Exception {
095        final byte[] ipBytes = new byte[size];
096        bufferm.get(ipBytes, bufferm.position(), size);
097        try {
098            return InetAddress.getByAddress(ipBytes);
099        } catch (UnknownHostException e) {
100            throw new Dhcp6Exception("Unable to parse inet address from: " + BaseEncoding.base16().encode(ipBytes), e);
101        }
102    }
103
104    static boolean isRelayedMessage(final Dhcp6MessageType type) {
105        return type == Dhcp6MessageType.DHCP_RELAY_FORW || type == Dhcp6MessageType.DHCP_RELAY_REPL;
106    }
107
108    /**
109     * https://tools.ietf.org/html/rfc3315#section-15.1
110     */
111    private static int getTxId(final ByteBuffer buffer) {
112        final byte[] txBytes = new byte[4];
113        buffer.get(txBytes, 1, 3);
114        txBytes[0] = 0;
115        return Ints.fromByteArray(txBytes);
116    }
117
118    /**
119     * https://tools.ietf.org/html/rfc3315#section-5.3
120     */
121    @Nonnull
122    private static Dhcp6MessageType getMsgType(final ByteBuffer buffer) throws Dhcp6Exception.UnknownMsgException {
123        final byte type = getByte(buffer);
124        final Dhcp6MessageType dhcp6MessageType = Dhcp6MessageType.forTypeCode(type);
125        Dhcp6Exception.UnknownMsgException.check(dhcp6MessageType, type);
126        return dhcp6MessageType;
127    }
128
129    private static byte getByte(final ByteBuffer buffer) {
130        return buffer.get();
131    }
132
133    @Nonnull
134    private static byte[] decodeBytes(@Nonnull ByteBuffer buffer, @Nonnegative int len) {
135        byte[] bytes = new byte[len];
136        buffer.get(bytes);
137        return bytes;
138    }
139
140    /**
141     * https://tools.ietf.org/html/rfc3315#section-5.3
142     */
143    @Nonnull
144    public Dhcp6Options decodeOptions(@Nonnull ByteBuffer message) throws Dhcp6Exception {
145        Dhcp6Options options = new Dhcp6Options();
146
147        while (message.hasRemaining()) {
148            short tag = message.getShort();
149            LOG.trace("Option type: {}", tag);
150            short length = message.getShort();
151            LOG.trace("Option len: {}", length);
152            byte[] value = decodeBytes(message, length);
153            final Dhcp6Option option = newOptionInstance(tag, value);
154            LOG.trace("Option: {}", option);
155            options.add(option);
156        }
157
158        return options;
159    }
160
161    @Nonnull
162    private Dhcp6Option newOptionInstance(@Nonnegative short tag, @Nonnull byte[] value) throws Dhcp6Exception {
163        Class<? extends Dhcp6Option> type = registry.getOptionType(tag);
164        Dhcp6Option option = (type != null) ? Dhcp6OptionsRegistry.newInstance(type) : new UnrecognizedOption(tag);
165        option.setData(value);
166        return option;
167    }
168}
