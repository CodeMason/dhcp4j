001package org.anarres.dhcp.v6.options;
002
003import java.nio.ByteBuffer;
004import java.util.HashMap;
005import java.util.Map;
006import javax.annotation.Nonnull;
007import org.anarres.dhcp.v6.io.Dhcp6MessageDecoder;
008import org.anarres.dhcp.v6.io.Dhcp6MessageEncoder;
009import org.apache.directory.server.dhcp.DhcpException;
010
011/**
012 * https://tools.ietf.org/html/rfc3315#section-22.17
013 */
014public class VendorSpecificInformationOption extends SuboptionOption {
015
016    private static final short TAG = 17;
017    private static final int HEADER_SIZE = 4;
018
019    /**
020     * Default decoder treats all suboptions as unknown
021     */
022    public static final Dhcp6MessageDecoder DEFAULT_DECODER = new Dhcp6MessageDecoder(new Dhcp6OptionsRegistry());
023    public static final Map<Integer, Dhcp6MessageDecoder> DECODERS = new HashMap<>();
024
025    /**
026     *
027     * Add vendor specific suboption decoder
028     *
029     * @param enterpriseNumber
030     * @param dhcp6MessageDecoder
031     */
032    public static void addDecoder(final int enterpriseNumber, final Dhcp6MessageDecoder dhcp6MessageDecoder) {
033        // TODO enable multiple option type registries per vendor
034        DECODERS.put(enterpriseNumber, dhcp6MessageDecoder);
035    }
036
037    public void setEnterpriseNumber(int enterpriseNumber) {
038        ByteBuffer buf = ByteBuffer.wrap(getData());
039        buf.putInt(0, enterpriseNumber);
040    }
041
042    public int getEnterpriseNumber() {
043        ByteBuffer buf = ByteBuffer.wrap(getData());
044        return buf.getInt(0);
045    }
046
047    @Override
048    public int getHeaderSize() {
049        return HEADER_SIZE;
050    }
051
052    @Override
053    public short getTag() {
054        return TAG;
055    }
056
057    @Override
058    protected Dhcp6MessageDecoder getDecoder() {
059        final Dhcp6MessageDecoder specificDecoder = DECODERS.get(getEnterpriseNumber());
060        return specificDecoder == null ? DEFAULT_DECODER : specificDecoder;
061    }
062
063    @Override
064    public String toString() {
065        final StringBuilder values = new StringBuilder();
066        values.append("Enterprise number:");
067        values.append(getEnterpriseNumber());
068        values.append(", ");
069        try {
070            values.append(getOptions().toString());
071        } catch (DhcpException e) {
072            values.append("options:[");
073            values.append(toStringDataFallback(getOptionsRaw().array()));
074            values.append("]");
075        }
076
077        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
078    }
079
080    public static VendorSpecificInformationOption create(@Nonnull final int enterpriseNumber, @Nonnull final Dhcp6Options options) {
081        final VendorSpecificInformationOption iaNaOption = new VendorSpecificInformationOption();
082        int length = HEADER_SIZE;
083
084        ByteBuffer encodedOptions = null;
085        if (!options.isEmpty()) {
086            encodedOptions = Dhcp6MessageEncoder.getInstance().encode(options);
087            length += encodedOptions.limit();
088        }
089
090        iaNaOption.setData(new byte[length]);
091        iaNaOption.setEnterpriseNumber(enterpriseNumber);
092        if (encodedOptions != null) {
093            iaNaOption.setOptions(encodedOptions);
094        }
095        return iaNaOption;
096    }
097
098}
