005package org.anarres.dhcp.v6.options;
006
007import com.google.common.base.Charsets;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010
011/**
012 * https://tools.ietf.org/html/rfc3315#section-22.13
013 * https://tools.ietf.org/html/rfc3315#section-24.4
014 *
015 * @author marosmars
016 */
017public class StatusCodeOption extends Dhcp6Option {
018
019    private static final short TAG = 13;
020
021    /**
022     * Success
023     */
024    public static final short SUCCESS = 0;
025
026    /**
027     * Failure, reason unspecified; this status code is sent by either a client or a server
028     * to indicate a failure not explicitly specified in this document.
029     */
030    public static final short UNSPEC_FAIL = 1;
031
032    /**
033     * Server has no addresses available to assign to the IA(s).
034     */
035    public static final short NO_ADDRS_AVAIL = 2;
036
037    /**
038     * Client record (binding) unavailable.
039     */
040    public static final short NO_BINDING = 3;
041
042    /**
043     * The prefix for the address is not appropriate for the link to which the client is attached.
044     */
045    public static final short NOT_ON_LINK = 4;
046
047    /**
048     * Sent by a server to a client to force the client to send messages to the server.
049     * using the All_DHCP_Relay_Agents_and_Servers address.
050     */
051    public static final short USE_MULTICAST = 5;
052
053
054    @Override
055    public short getTag() {
056        return TAG;
057    }
058
059    public short getStatusCode() {
060        ByteBuffer buf = ByteBuffer.wrap(getData());
061        return buf.getShort();
062    }
063
064    public void setStatusCode(short statusCode) {
065        ByteBuffer buf = ByteBuffer.wrap(getData());
066        buf.putShort(0, statusCode);
067    }
068
069    @Nonnull public String getMessage() {
070        return new String(getData(), 2, getData().length - 2, Charsets.UTF_8);
071    }
072
073    public void setMessage(@Nonnull String message) {
074        ByteBuffer buf = ByteBuffer.wrap(getData());
075        buf.position(2);
076        buf.put(message.getBytes(Charsets.UTF_8));
077    }
078
079    public void setPreference(byte preference) {
080        setData(new byte[]{preference});
081    }
082
083    @Override
084    public String toString() {
085        final StringBuilder values = new StringBuilder();
086        values.append("Code:");
087        values.append(getStatusCode());
088        values.append(", Message:");
089        values.append(getMessage());
090        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
091    }
092
093    @Nonnull
094    public static StatusCodeOption create(final short statusCode, @Nonnull final String message) {
095        final StatusCodeOption option = new StatusCodeOption();
096        option.setData(new byte[2 + message.getBytes(Charsets.UTF_8).length]);
097        option.setStatusCode(statusCode);
098        option.setMessage(message);
099        return option;
100    }
101
102    @Nonnull
103    public static Dhcp6Option create(final short statusCode) {
104        final StatusCodeOption option = new StatusCodeOption();
105        option.setData(new byte[2]);
106        option.setStatusCode(statusCode);
107        return option;
108    }
109}
