005package org.anarres.dhcp.v6.options;
006
007import com.google.common.base.Preconditions;
008import java.net.Inet6Address;
009import java.net.InetAddress;
010import java.net.UnknownHostException;
011import java.nio.ByteBuffer;
012import javax.annotation.Nonnull;
013
014/**
015 * https://tools.ietf.org/html/rfc3315#section-22.12
016 *
017 * @author marosmars
018 */
019public class ServerUnicastOption extends Dhcp6Option {
020
021    private static final short TAG = 12;
022
023    @Override
024    public short getTag() {
025        return TAG;
026    }
027
028    @Nonnull
029    public InetAddress getIp() {
030        ByteBuffer buf = ByteBuffer.wrap(getData());
031        final byte[] dst = new byte[16];
032        buf.get(dst, 0, 16);
033
034        try {
035            return Inet6Address.getByAddress(dst);
036        } catch (UnknownHostException e) {
037            // This can happen only if the IP byte array is of illegal length
038            throw new IllegalStateException("Illegal IP address", e);
039        }
040    }
041
042    public void setIp(@Nonnull final InetAddress address) {
043        Preconditions.checkArgument(address instanceof Inet6Address, "Ipv4 detected: %s", address);
044        ByteBuffer buf = ByteBuffer.wrap(getData());
045        buf.put(address.getAddress());
046    }
047
048    @Override
049    public String toString() {
050        final StringBuilder values = new StringBuilder();
051        values.append("IP:");
052        values.append(getIp());
053        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
054    }
055
056    public static ServerUnicastOption create(@Nonnull final InetAddress ip) {
057        final ServerUnicastOption iaNaOption = new ServerUnicastOption();
058        iaNaOption.setData(new byte[16]);
059        iaNaOption.setIp(ip);
060        return iaNaOption;
061    }
062}
