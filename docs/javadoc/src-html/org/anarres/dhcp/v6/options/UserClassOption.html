001package org.anarres.dhcp.v6.options;
002
003import javax.annotation.Nonnull;
004
005/**
006 * https://tools.ietf.org/html/rfc3315#section-22.15
007 */
008public class UserClassOption extends Dhcp6Option {
009    private static final short TAG = 15;
010
011    @Override public short getTag() {
012        return TAG;
013    }
014
015    @Nonnull
016    public static UserClassOption create(@Nonnull final byte[] data) {
017        final UserClassOption relayMessageOption = new UserClassOption();
018        relayMessageOption.setData(data);
019        return relayMessageOption;
020    }
021
022}
