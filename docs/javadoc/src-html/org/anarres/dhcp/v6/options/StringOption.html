001package org.anarres.dhcp.v6.options;
002
003import com.google.common.base.Charsets;
004
005/**
006 * Base option for string based options
007 */
008public abstract class StringOption extends Dhcp6Option {
009
010    public String getString() {
011        return new String(getData(), Charsets.UTF_8);
012    }
013
014    public void setString(final String value) {
015        setData(value.getBytes(Charsets.UTF_8));
016    }
017
018    @Override public String toString() {
019        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + getString();
020    }
021}
