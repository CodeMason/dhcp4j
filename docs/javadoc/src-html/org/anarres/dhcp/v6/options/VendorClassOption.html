001package org.anarres.dhcp.v6.options;
002
003import com.google.common.io.BaseEncoding;
004import java.nio.ByteBuffer;
005import javax.annotation.Nonnull;
006
007/**
008 * https://tools.ietf.org/html/rfc3315#section-22.16
009 */
010public class VendorClassOption extends Dhcp6Option {
011    private static final short TAG = 16;
012
013    @Override public short getTag() {
014        return TAG;
015    }
016
017    public void setEnterpriseNumber(int enterpriseNumber) {
018        ByteBuffer buf = ByteBuffer.wrap(getData());
019        buf.putInt(0, enterpriseNumber);
020    }
021
022    public int getEnterpriseNumber() {
023        ByteBuffer buf = ByteBuffer.wrap(getData());
024        return buf.getInt(0);
025    }
026
027    @Nonnull
028    public byte[] getVendorClassData() {
029        ByteBuffer buf = ByteBuffer.wrap(getData());
030        final byte[] vendorClassData = new byte[getData().length - 4];
031        buf.get(vendorClassData, 4, getData().length);
032        return vendorClassData;
033    }
034
035    public void setVendorClassData(@Nonnull final byte[] data) {
036        ByteBuffer buf = ByteBuffer.wrap(getData());
037        buf.put(data, 4, data.length);
038    }
039
040    @Override
041    public String toString() {
042        final StringBuilder values = new StringBuilder();
043        values.append("enterpriseNumber:");
044        values.append(getEnterpriseNumber());
045        values.append(", vendorClassData:");
046        values.append(BaseEncoding.base16().encode(getVendorClassData()));
047        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
048    }
049
050    public static VendorClassOption create(final int enterpriseNumber, final byte[] vendorClassData) {
051        final VendorClassOption iaNaOption = new VendorClassOption();
052        iaNaOption.setData(new byte[vendorClassData.length + 4]);
053        iaNaOption.setEnterpriseNumber(enterpriseNumber);
054        iaNaOption.setVendorClassData(vendorClassData);
055        return iaNaOption;
056    }
057}
