005package org.anarres.dhcp.v6.options;
006
007import java.nio.ByteBuffer;
008import javax.annotation.Nonnull;
009import org.anarres.dhcp.v6.io.Dhcp6MessageDecoder;
010import org.apache.directory.server.dhcp.DhcpException;
011
012/**
013 * http://tools.ietf.org/html/rfc3315#section-22.4
014 *
015 * @author shevek
016 */
017public class IaNaOption extends Dhcp6Option {
018
019    private static final short TAG = 3;
020
021    @Override
022    public short getTag() {
023        return TAG;
024    }
025
026    public int getIAID() {
027        ByteBuffer buf = ByteBuffer.wrap(getData());
028        return buf.getInt(0);
029    }
030
031    public void setIAID(int IAID) {
032        ByteBuffer buf = ByteBuffer.wrap(getData());
033        buf.putInt(0, IAID);
034    }
035
036    public int getT1() {
037        ByteBuffer buf = ByteBuffer.wrap(getData());
038        return buf.getInt(4);
039    }
040
041    public void setT1(int t1) {
042        ByteBuffer buf = ByteBuffer.wrap(getData());
043        buf.putInt(4, t1);
044    }
045
046    public int getT2() {
047        ByteBuffer buf = ByteBuffer.wrap(getData());
048        return buf.getInt(8);
049    }
050
051    @Nonnull
052    public Dhcp6Options getOptions() throws DhcpException {
053        byte[] data = getData();
054        ByteBuffer buf = ByteBuffer.wrap(data, 12, data.length - 12);
055        Dhcp6MessageDecoder decoder = new Dhcp6MessageDecoder();
056        return decoder.decodeOptions(buf);
057    }
058}
