001package org.anarres.dhcp.v6.options;
002
003import com.google.common.collect.Iterables;
004import com.google.common.collect.Lists;
005import java.nio.ByteBuffer;
006import java.util.List;
007
008/**
009 * https://tools.ietf.org/html/rfc3315#section-22.7
010 */
011public class OptionRequestOption extends Dhcp6Option {
012
013    private static final short TAG = 6;
014
015    @Override
016    public short getTag() {
017        return TAG;
018    }
019
020    public Iterable<Short> getRequestedOptions() {
021        ByteBuffer buf = ByteBuffer.wrap(getData());
022        List<Short> options = Lists.newArrayListWithCapacity(getData().length / 2);
023
024        while(buf.remaining()!=0) {
025            options.add(buf.getShort());
026        }
027
028        return options;
029    }
030
031    @Override
032    public String toString() {
033        final StringBuilder values = new StringBuilder();
034        try {
035            values.append("requestedOptions:[");
036            values.append(Iterables.toString(getRequestedOptions()));
037            values.append("]");
038        } catch (RuntimeException e) {
039            values.append("requestedOptions:[");
040            values.append(toStringDataFallback(getData()));
041            values.append("]");
042        }
043
044        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
045    }
046}
