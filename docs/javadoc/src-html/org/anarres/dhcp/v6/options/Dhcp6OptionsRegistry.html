006package org.anarres.dhcp.v6.options;
007
008import com.google.common.collect.BiMap;
009import com.google.common.collect.HashBiMap;
010import javax.annotation.CheckForNull;
011import javax.annotation.Nonnull;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * See http://www.iana.org/assignments/bootp-dhcp-parameters/bootp-dhcp-parameters.xhtml#options
017 *
018 * @author shevek
019 * @author marosmars
020 * @author marekgr
021 */
022public class Dhcp6OptionsRegistry {
023
024    private static final Logger LOG = LoggerFactory.getLogger(Dhcp6OptionsRegistry.class);
025
026    @SuppressWarnings("unchecked")
027    private static class Inner {
028
029        private static final Dhcp6OptionsRegistry INSTANCE = new Dhcp6OptionsRegistry();
030
031        @SuppressWarnings("rawtypes")
032        private static final Class OPTION_CLASSES[] = { IaNaOption.class, ClientIdOption.class, ServerIdOption.class,
033                        ElapsedTimeOption.class, IaTaOption.class, IaAddressOption.class, NewPOSIXTimezone.class,
034                        NewTZDBTimezone.class, OptionRequestOption.class, StatusCodeOption.class,
035                        RelayMessageOption.class, PreferenceOption.class, InterfaceIdOption.class,
036                        ServerUnicastOption.class, UserClassOption.class, VendorClassOption.class,
037                        VendorSpecificInformationOption.class };
038
039        static {
040            for (Class<? extends Dhcp6Option> optionType : OPTION_CLASSES) {
041                INSTANCE.addOptionType(optionType);
042            }
043        }
044
045    }
046
047    @Nonnull
048    public static Dhcp6OptionsRegistry getInstance() {
049        return Inner.INSTANCE;
050    }
051
052    private final BiMap<Short, Class<? extends Dhcp6Option>> optionTypes = HashBiMap.create();
053
054    @Nonnull
055    public static <T extends Dhcp6Option> T newInstance(@Nonnull Class<T> type) {
056        try {
057            return type.newInstance();
058        } catch (InstantiationException | IllegalAccessException e) {
059            throw new IllegalArgumentException("Cannot instantiate " + type, e);
060        }
061    }
062
063    @Nonnull
064    public static <T extends Dhcp6Option> T copy(@Nonnull Class<T> type, @Nonnull T option) {
065        final T t = newInstance(type);
066        t.setData(option.getData());
067        return t;
068    }
069
070    private short getTagFrom(@Nonnull Class<? extends Dhcp6Option> type) {
071        Dhcp6Option o = newInstance(type);
072        return o.getTag();
073    }
074
075    public void addOptionType(@Nonnull Class<? extends Dhcp6Option> type) {
076        short tag = getTagFrom(type);
077        if (optionTypes.put(tag, type) != null)
078            throw new IllegalArgumentException("Duplicate tag: " + type);
079    }
080
081    @CheckForNull
082    public Class<? extends Dhcp6Option> getOptionType(short tag) {
083        return optionTypes.get(tag);
084    }
085
086    @Nonnull
087    public short getOptionTag(@Nonnull Class<? extends Dhcp6Option> type) {
088        Short tag = optionTypes.inverse().get(type);
089        if (tag != null)
090            return tag;
091        LOG.warn("Unregistered option type: {}", type);
092        return getTagFrom(type);
093    }
094
095    @Override
096    public String toString() {
097        return getClass().getSimpleName() + "(" + optionTypes + ")";
098    }
099
100}
