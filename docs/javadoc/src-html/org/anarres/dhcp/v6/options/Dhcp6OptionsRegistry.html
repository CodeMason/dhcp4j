006package org.anarres.dhcp.v6.options;
007
008import com.google.common.collect.BiMap;
009import com.google.common.collect.HashBiMap;
010import javax.annotation.CheckForNull;
011import javax.annotation.Nonnull;
012
013/**
014 * See http://www.iana.org/assignments/bootp-dhcp-parameters/bootp-dhcp-parameters.xhtml#options
015 *
016 * @author shevek
017 */
018public class Dhcp6OptionsRegistry {
019
020    private static class Inner {
021
022        private static final Dhcp6OptionsRegistry INSTANCE = new Dhcp6OptionsRegistry();
023
024        private static final Class OPTION_CLASSES[] = {
025
026        };
027
028        static {
029            for (Class<? extends Dhcp6Option> optionType : OPTION_CLASSES) {
030                INSTANCE.addOptionType(optionType);
031            }
032        }
033
034    }
035
036    @Nonnull
037    public static Dhcp6OptionsRegistry getInstance() {
038        return Inner.INSTANCE;
039    }
040
041    private final BiMap<Short, Class<? extends Dhcp6Option>> optionTypes = HashBiMap.create();
042
043    @Nonnull
044    public static <T extends Dhcp6Option> T newInstance(@Nonnull Class<T> type) {
045        try {
046            return type.newInstance();
047        } catch (InstantiationException e) {
048            throw new IllegalArgumentException("Cannot instantiate " + type, e);
049        } catch (IllegalAccessException e) {
050            throw new IllegalArgumentException("Cannot instantiate " + type, e);
051        }
052    }
053
054    private short getTagFrom(@Nonnull Class<? extends Dhcp6Option> type) {
055        Dhcp6Option o = newInstance(type);
056        return o.getTag();
057    }
058
059    public void addOptionType(@Nonnull Class<? extends Dhcp6Option> type) {
060        short tag = getTagFrom(type);
061        if (optionTypes.put(tag, type) != null)
062            throw new IllegalArgumentException("Duplicate tag: " + type);
063    }
064
065    @CheckForNull
066    public Class<? extends Dhcp6Option> getOptionType(short tag) {
067        return optionTypes.get(tag);
068    }
069
070    @Nonnull
071    public short getOptionTag(@Nonnull Class<? extends Dhcp6Option> type) {
072        Short tag = optionTypes.inverse().get(type);
073        if (tag != null)
074            return tag;
075        // TODO: Warn about unregistered option.
076        return getTagFrom(type);
077    }
078
079    @Override
080    public String toString() {
081        return getClass().getSimpleName() + "(" + optionTypes + ")";
082    }
083
084}
