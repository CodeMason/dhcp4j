005package org.anarres.dhcp.v6.options;
006
007import org.apache.directory.server.dhcp.DhcpException;
008
009/**
010 * http://tools.ietf.org/html/rfc3315#section-22.5
011 *
012 * @author marosmars
013 */
014public class IaTaOption extends IaOption {
015
016    private static final short TAG = 4;
017    private static final int HEADER_LENGTH = 12;
018
019    @Override
020    public short getTag() {
021        return TAG;
022    }
023
024    @Override
025    public String toString() {
026        final StringBuilder values = new StringBuilder();
027        values.append("IAID:");
028        values.append(getIAID());
029        values.append(", ");
030        try {
031            values.append(getOptions().toString());
032        } catch (DhcpException e) {
033            values.append("options:[");
034            values.append(toStringDataFallback(getOptionsRaw().array()));
035            values.append("]");
036        }
037
038        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
039    }
040
041    @Override
042    public int getHeaderSize() {
043        return HEADER_LENGTH;
044    }
045}
