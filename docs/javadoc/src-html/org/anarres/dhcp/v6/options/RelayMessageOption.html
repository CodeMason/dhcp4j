001package org.anarres.dhcp.v6.options;
002
003import com.google.common.io.BaseEncoding;
004import java.io.IOException;
005import java.nio.ByteBuffer;
006import javax.annotation.Nonnull;
007import org.anarres.dhcp.v6.Dhcp6Exception;
008import org.anarres.dhcp.v6.io.Dhcp6MessageDecoder;
009import org.anarres.dhcp.v6.io.Dhcp6MessageEncoder;
010import org.anarres.dhcp.v6.messages.Dhcp6Message;
011
012/**
013 * https://tools.ietf.org/html/rfc3315#section-22.10
014 */
015public class RelayMessageOption extends Dhcp6Option {
016    private static final short TAG = 9;
017
018    @Override public short getTag() {
019        return TAG;
020    }
021
022    @Nonnull
023    public Dhcp6Message getRelayedMessage() throws Dhcp6Exception {
024        try {
025            return Dhcp6MessageDecoder.getInstance().decode(ByteBuffer.wrap(getData()));
026        } catch (IOException e) {
027            throw new Dhcp6Exception("Unable to decode relayed dhcpv6 message: " + BaseEncoding.base16().encode(getData()), e);
028        }
029    }
030
031    public void setRelayedMessage(@Nonnull final Dhcp6Message message) {
032        Dhcp6MessageEncoder.getInstance().encode(ByteBuffer.wrap(getData()), message);
033    }
034
035    @Override
036    public String toString() {
037        final StringBuilder values = new StringBuilder();
038        try {
039            values.append("requestedOptions:[");
040            values.append(getRelayedMessage().toString());
041            values.append("]");
042        } catch (Dhcp6Exception e) {
043            values.append("requestedOptions:[");
044            values.append(toStringDataFallback(getData()));
045            values.append("]");
046        }
047
048        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
049    }
050
051    @Nonnull
052    public static RelayMessageOption create(@Nonnull final Dhcp6Message relayedMessage) {
053        final byte[] data = new byte[relayedMessage.getLength()];
054        final RelayMessageOption relayMessageOption = new RelayMessageOption();
055        relayMessageOption.setData(data);
056        relayMessageOption.setRelayedMessage(relayedMessage);
057        return relayMessageOption;
058    }
059
060}
