077    private <T extends IaOption> void leaseIA(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg,
078        final @Nonnull Dhcp6Message reply, Class<T> iaType, final ClientBindingRegistry iaRegistry)
079        throws Dhcp6Exception {
080
081        final DuidOption.Duid clientId = getClientId(incomingMsg);
082        for (T incomingIaOption : incomingMsg.getOptions().getAll(iaType)) {
083            LOG.debug("Client {} requested IA:{} for {}. IA option: {}", clientId, iaRegistry,
084                            incomingIaOption.getIAID(), incomingIaOption);
085
086            final Dhcp6Options iaResponseOptions = new Dhcp6Options();
087
088            InetAddress ip = getIp(requestContext, clientId, incomingIaOption, iaRegistry);
089            if (ip != null) {
090                // For any IAs to which the server can assign addresses, the server
091                // includes the IA with addresses and other configuration parameters,
092                // and records the IA as a new client binding.
093                final IaAddressOption option = wrapIp(clientId, incomingIaOption, iaRegistry,
094                                getIp(requestContext, clientId, incomingIaOption, iaRegistry));
095                iaResponseOptions.add(option);
096            } else {
097                // If the server cannot assign any addresses to an IA in the message
098                // from the client, the server MUST include the IA in the Reply message
099                // with no addresses in the IA and a Status Code option in the IA
100                // containing status code NoAddrsAvail.
101                iaResponseOptions.add(StatusCodeOption.create(StatusCodeOption.NO_ADDRS_AVAIL));
102            }
103            final T copy = createIaOption(iaType, incomingIaOption, iaResponseOptions);
104
105            reply.getOptions().add(copy);
106        }
107    }
108
109    private <T extends IaOption> T createIaOption(final Class<T> iaType, final T incomingIaOption, final Dhcp6Options iaResponseOptions) {
110        final T copy = Dhcp6OptionsRegistry.newInstance(iaType);
111        copy.setData(new byte[copy.getHeaderSize() + iaResponseOptions.getLength()]);
112        setTimeParameters(copy);
113        copy.setIAID(incomingIaOption.getIAID());
114        copy.setOptions(iaResponseOptions);
115        return copy;
116    }

131    protected final <T extends IaOption> void renewIA(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg,
132        final Dhcp6Message reply, final Class<T> iaType, final ClientBindingRegistry iaRegistry)
133        throws Dhcp6Exception {
134
135        final DuidOption.Duid clientId = getClientId(incomingMsg);
136        for (T incomingIaOption : incomingMsg.getOptions().getAll(iaType)) {
137
138            final Dhcp6Options iaResponseOptions = new Dhcp6Options();
139            final Dhcp6Option option;
140
141            if (iaRegistry.contains(clientId, incomingIaOption.getIAID())) {
142                option = renewIAInstance(requestContext, iaRegistry, clientId, incomingIaOption, "renews");
143            } else {
144                LOG.debug("Client {} renews UNKNOWN IA:{} for {}. IA option: {}", clientId, iaRegistry,
145                                incomingIaOption.getIAID(), incomingIaOption);
146                option = StatusCodeOption.create(StatusCodeOption.NO_BINDING);
147            }
148
149            iaResponseOptions.add(option);
150            final T copy = createIaOption(iaType, incomingIaOption, iaResponseOptions);
151
152            reply.getOptions().add(copy);
153        }
154    }

163    private <T extends IaOption> Dhcp6Option renewIAInstance(final Dhcp6RequestContext requestContext,
164        final ClientBindingRegistry iaRegistry, final DuidOption.Duid clientId, final T incomingIaOption, final String type)
165        throws Dhcp6Exception {
166
167        final Dhcp6Option option;
168        LOG.debug("Client {} {} IA:{} for {}. IA option: {}", clientId, type, iaRegistry, incomingIaOption.getIAID(), incomingIaOption);
169        // FIXME what if the address from client does not match address stored for the client on server ?
170        // should it return NO_BINDING or UNSPEC_FAIL?
171        final InetAddress ip = iaRegistry.get(clientId, incomingIaOption.getIAID()).getIp();
172
173        final Iterable<InetAddress> addressesFromIa = getAddressesFromIa(incomingIaOption);
174
175        // Check whether client provided at least one address for IA
176        if (Iterables.isEmpty(addressesFromIa)) {
177            LOG.warn("Client {} {} IA:{} for {}. IA option: {}. No address present", clientId, type, iaRegistry,
178                            incomingIaOption.getIAID(), incomingIaOption);
179        }
180
181        // FIXME can the client request renew with a single IaNa option containing multiple addresses ? according to RFC
182        // this is possible
183        // but it is real ? how would client have multiple addresses for a single Ia ?
184        // for example see http://serverfault.com/questions/312221/can-a-single-network-card-have-2-ip-addresses
185
186        // Check whether client provided the same address as is assigned to it
187        final InetAddress ipFromClient = Iterables.getFirst(addressesFromIa, null);
188        if (!ipFromClient.equals(ip)) {
189            LOG.warn("Client {} {} UNKNOWN address:{} vs {}, IA:{} for {}.", clientId, type, ipFromClient, ip,
190                            iaRegistry, incomingIaOption.getIAID());
191        }
192
193        option = wrapIp(clientId, incomingIaOption, iaRegistry, ip);
194
195        if(!isAppropriate(requestContext, clientId, incomingIaOption.getIAID(), ip)) {
196            LOG.debug("Client {} IA:{} for {}. Inappropriate address detected: {}", clientId, iaRegistry, incomingIaOption.getIAID(), ip);
197            ((IaAddressOption) option).setPreferredLifetime(0);
198            ((IaAddressOption) option).setValidLifetime(0);
199        }
200        return option;
201    }

223    protected final <T extends IaOption> void rebindIA(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg,
224        final Dhcp6Message reply, final Class<T> iaType, final ClientBindingRegistry iaRegistry) throws Dhcp6Exception {
225        final DuidOption.Duid clientId = getClientId(incomingMsg);
226        for (T incomingIaOption : incomingMsg.getOptions().getAll(iaType)) {
227
228            final Dhcp6Options iaResponseOptions = new Dhcp6Options();
229            final Dhcp6Option option;
230
231            if (iaRegistry.contains(clientId, incomingIaOption.getIAID())) {
232                option = renewIAInstance(requestContext, iaRegistry, clientId, incomingIaOption, "rebinds");
233            } else {
234                LOG.debug("Client {} rebinds UNKNOWN IA:{} for {}. IA option: {}", clientId, iaRegistry, incomingIaOption.getIAID(), incomingIaOption);
235                throw new Dhcp6Exception.UnableToAnswerException("No IA found for " + incomingIaOption.getIAID());
236            }
237
238            iaResponseOptions.add(option);
239            final T copy = createIaOption(iaType, incomingIaOption, iaResponseOptions);
240
241            reply.getOptions().add(copy);
242        }
243    }

258    protected final <T extends IaOption> List<ClientBindingRegistry.ClientBinding> releaseIa(final Dhcp6Message incomingMsg,
259        final Dhcp6Message reply, final Class<T> iaType, final ClientBindingRegistry iaRegistry) {
260        final DuidOption.Duid clientId = getClientId(incomingMsg);
261
262        List<ClientBindingRegistry.ClientBinding> removedBindings = new ArrayList<>();
263
264        for (T incomingIaOption : incomingMsg.getOptions().getAll(iaType)) {
265            LOG.debug("Client {} trying to release IA:{} for {}. Ia option: {}", clientId, iaRegistry, incomingIaOption.getIAID(), incomingIaOption);
266
267            if (iaRegistry.contains(clientId, incomingIaOption.getIAID())) {
268                removedBindings.add(iaRegistry.remove(clientId, incomingIaOption.getIAID()));
269                LOG.debug("Client {} released IA:{} successfully {}", clientId, iaRegistry, incomingIaOption.getIAID());
270            } else {
271                LOG.warn("Client {} tried to release unknown IA:{}: {}", clientId, iaRegistry, incomingIaOption.getIAID());
272                // TODO make this options a constant
273                final Dhcp6Options unknownIaOptions = new Dhcp6Options();
274                unknownIaOptions.add(StatusCodeOption.create(StatusCodeOption.NO_BINDING));
275
276                final T copy = createIaOption(iaType, incomingIaOption, unknownIaOptions);
277
278                reply.getOptions().add(copy);
279            }
280        }
281
282        return removedBindings;
283    }

328    public Dhcp6Message confirm(final Dhcp6RequestContext requestContext,
329        final Dhcp6Message incomingMsg, final Dhcp6Message reply) throws Dhcp6Exception {
330        try {
331            final short confirmedStatus = confirmIAs(requestContext, incomingMsg) ?
332                StatusCodeOption.SUCCESS :
333                StatusCodeOption.NOT_ON_LINK;
334
335            reply.getOptions().add(StatusCodeOption.create(confirmedStatus));
336        } catch (Dhcp6Exception.UnableToAnswerException e) {
337            LOG.warn("Skipping request {}, unable to confirm", incomingMsg, e);
338            return null;
339        }
340
341        requestInformation(requestContext, incomingMsg, reply);
342        return reply;
343    }

345    /**
346     * Confirm all IA groups in the confirm message. Base implementation confirms IaNa and IaTa.
347     */
348    protected boolean confirmIAs(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg)
349        throws Dhcp6Exception {
350        boolean confirmed = confirmIa(requestContext, incomingMsg, IaNaOption.class, iaNaRegistry);
351        confirmed &= confirmIa(requestContext, incomingMsg, IaTaOption.class, iaTaRegistry);
352        return confirmed;
353    }

355    protected final <T extends IaOption> boolean confirmIa(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg,
356       final Class<T> iaType, final ClientBindingRegistry iaRegistry) throws Dhcp6Exception {
357
358        final DuidOption.Duid clientId = getClientId(incomingMsg);
359
360        for (IaOption incomingIaOption : incomingMsg.getOptions().getAll(iaType)) {
361            LOG.debug("Client {} confirming IA:{} for {}. IA option: {}", clientId, iaRegistry, incomingIaOption.getIAID(), incomingIaOption);
362
363            if (iaRegistry.contains(clientId, incomingIaOption.getIAID())) {
364                final Iterable<InetAddress> addresses = getAddressesFromIa(incomingIaOption);
365
366                //or there were no addresses in any of the IAs sent by the client, the server MUST NOT send a reply to the client.
367                if(Iterables.isEmpty(addresses)) {
368                    LOG.warn("Client {} confirming IA:{} for {} with no address. Unable to answer", clientId, iaRegistry, incomingIaOption.getIAID());
369                    throw new Dhcp6Exception.UnableToAnswerException("No addresses for " + incomingIaOption.getIAID());
370                }
371
372                for (InetAddress address : addresses) {
373                    final boolean appropriate = isAppropriate(requestContext, clientId, incomingIaOption.getIAID(), address);
374                    if(!appropriate) {
375                        LOG.debug("Client {} inappropriate address detected for IA: {}, IA optiona: {}", clientId, incomingIaOption.getIAID(), incomingIaOption);
376                        return false;
377                    }
378                }
379            }
380
381        }
382
383        return true;
384    }

386    public static Iterable<InetAddress> getAddressesFromIa(final SuboptionOption incomingIaNaOption) throws Dhcp6Exception {
387        return Iterables.transform(incomingIaNaOption.getOptions().getAll(IaAddressOption.class), new Function<IaAddressOption, InetAddress>() {
388            @Override public InetAddress apply(final IaAddressOption input) {
389                return input.getIp();
390            }
391        });
392    }

398    private IaAddressOption wrapIp(final DuidOption.Duid clientId, final IaOption incomingIaOption,
399        final ClientBindingRegistry iaRegistry, final InetAddress ip) {
400
401        final IaAddressOption option = IaAddressOption
402            .create(ip, lifetimes.getPreferredLt(), lifetimes.getValidLt(), Optional.<Dhcp6Options>absent());
403
404        iaRegistry.add(clientId, incomingIaOption.getIAID(), ip);
405        LOG.debug("Client {} leased: {} for IaNa {}", clientId, ip, incomingIaOption.getIAID());
406        return option;
407    }

409    private InetAddress getIp(final Dhcp6RequestContext requestContext, final DuidOption.Duid clientId, final IaOption incomingIaNaOption,
410        final ClientBindingRegistry iaRegistry) throws Dhcp6Exception {
411        final InetAddress ip;
412        if (iaRegistry.contains(clientId, incomingIaNaOption.getIAID())) {
413            ip = iaRegistry.get(clientId, incomingIaNaOption.getIAID()).getIp();
414        } else {
415            ip = newIp(requestContext, clientId, incomingIaNaOption);
416        }
417        return ip;
418    }

428    /**
429     * Static lifetimes container
430     */
431    public static final class Lifetimes {
432
433        private final int t1, t2;
434        private final int preferredLt, validLt;
435
436        public Lifetimes(final int t1, final int t2, final int preferredLt, final int validLt) {
437            this.t1 = t1;
438            this.t2 = t2;
439            this.preferredLt = preferredLt;
440            this.validLt = validLt;
441            validate();
442        }
443
444        private void validate() {
445            Preconditions.checkArgument(getPreferredLt() < getValidLt(), "Preferred lifetime < Valid lifetime");
446            Preconditions.checkArgument(t1 < t2, "T1 < T2");
447            Preconditions.checkArgument(t1 < getPreferredLt(), "T1 < Preferred lifetime");
448            Preconditions.checkArgument(t2 < getPreferredLt(), "T2 < Preferred lifetime");
449        }
450
451        public int getPreferredLt() {
452            return preferredLt;
453        }
454
455        public int getT1() {
456            return t1;
457        }
458
459        public int getT2() {
460            return t2;
461        }
462
463        public int getValidLt() {
464            return validLt;
465        }
466    }
