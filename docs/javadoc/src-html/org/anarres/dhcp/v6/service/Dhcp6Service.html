001package org.anarres.dhcp.v6.service;
002
003import com.google.common.annotations.Beta;
004import com.google.common.base.Optional;
005import javax.annotation.CheckForNull;
006import javax.annotation.Nonnull;
007import org.anarres.dhcp.v6.Dhcp6Exception;
008import org.anarres.dhcp.v6.io.Dhcp6RequestContext;
009import org.anarres.dhcp.v6.messages.Dhcp6Message;
010
011/**
012 * DHCPv6 server side service responsible for handling DHCP client requests
013 */
014@Beta
015public interface Dhcp6Service {
016
017    /**
018     * Default DHCP client port
019     */
020    int CLIENT_PORT = 546;
021    /**
022     * Default DHCP server port
023     */
024    int SERVER_PORT = 547;
025
026    /**
027     * Try to reply to client's request. Its possible to return absent response, which means no response will be
028     * sent to the client
029     */
030    @CheckForNull
031    Optional<Dhcp6Message> getReplyFor(@Nonnull Dhcp6RequestContext requestContext, @Nonnull Dhcp6Message request)
032        throws Dhcp6Exception;
033}
