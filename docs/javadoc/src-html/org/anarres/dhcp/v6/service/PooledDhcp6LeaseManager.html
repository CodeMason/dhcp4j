001package org.anarres.dhcp.v6.service;
002
003import com.google.common.annotations.Beta;
004import com.google.common.net.InetAddresses;
005import java.net.InetAddress;
006import javax.annotation.Nonnull;
007import org.anarres.dhcp.v6.Dhcp6Exception;
008import org.anarres.dhcp.v6.io.Dhcp6RequestContext;
009import org.anarres.dhcp.v6.options.DuidOption;
010import org.anarres.dhcp.v6.options.IaOption;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014/**
015 * Basic implementation of a lease manager providing IPs from an IP pool
016 */
017@Beta
018public class PooledDhcp6LeaseManager extends AbstractDhcp6LeaseManager {
019
020    private static final Logger LOG = LoggerFactory.getLogger(PooledDhcp6LeaseManager.class);
021
022    private final InetAddress startingAddress;
023    private final InetAddress endingAddress;
024
025    // TODO should we use utilities from package org.anarres.dhcp.common.address here ??
026
027    public PooledDhcp6LeaseManager(@Nonnull final InetAddress startingAddress, @Nonnull final InetAddress endingAddress,
028        @Nonnull final Lifetimes lifetimes) {
029        super(lifetimes, ClientBindingRegistry.createForIaNa(), ClientBindingRegistry.createForIaTa());
030        this.startingAddress = startingAddress;
031        this.endingAddress = endingAddress;
032    }
033
034    /**
035     * Invoked during renew, rebind and confirm. Each address needs to be checked if still appropriate
036     *
037     * @throws org.anarres.dhcp.v6.Dhcp6Exception.UnableToAnswerException in case the test can not be performed
038     */
039    @Override protected boolean isAppropriate(final Dhcp6RequestContext requestContext, final DuidOption.Duid clientId, final int iaid,
040        final InetAddress ip) throws Dhcp6Exception.UnableToAnswerException {
041        return true;
042    }
043
044    /**
045     * @return next available IP
046     */
047    @Override protected InetAddress newIp(final Dhcp6RequestContext requestContext, final DuidOption.Duid clientId, final IaOption iaOption)
048        throws Dhcp6Exception {
049        // Client provided address hints
050        final Iterable<InetAddress> requestedAddresses = getAddressesFromIa(iaOption);
051        // TODO try to use hints in address allocation
052        
053        // Warning, this is highly inefficient
054        InetAddress current = startingAddress;
055        while (true) {
056            if(current.equals(endingAddress)) {
057                LOG.warn("IP pool exhausted");
058                return null;
059            }
060
061            if(!getIaNaRegistry().containsIp(current) && !getIaTaRegistry().containsIp(current)) {
062                return current;
063            }
064
065            current = InetAddresses.increment(current);
066        }
067    }
068
069}
