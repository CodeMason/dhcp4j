001package org.anarres.dhcp.v6.service;
002
003import com.google.common.annotations.Beta;
004import java.net.InetAddress;
005import java.util.HashMap;
006import java.util.HashSet;
007import java.util.Map;
008import java.util.Objects;
009import java.util.Set;
010import javax.annotation.Nonnull;
011import javax.annotation.Nullable;
012import javax.annotation.concurrent.GuardedBy;
013import javax.annotation.concurrent.ThreadSafe;
014import org.anarres.dhcp.v6.options.DuidOption;
015
016@Beta
017@ThreadSafe
018public class ClientBindingRegistry {
019
020    @GuardedBy("this")
021    private final Map<DuidOption.Duid, ClientBindings> bindings = new HashMap<>();
022    @GuardedBy("this")
023    private final Set<InetAddress> allIps = new HashSet<>();
024
025    private final String id;
026
027    public ClientBindingRegistry(@Nonnull final String id) {
028        this.id = id;
029    }
030
031    public synchronized void add(final DuidOption.Duid clientId, final int iaid, final InetAddress ip) {
032        if(!bindings.containsKey(clientId)) {
033            bindings.put(clientId, new ClientBindings());
034        }
035
036        final ClientBindings clientBindings = bindings.get(clientId);
037        clientBindings.add(new ClientBinding(iaid, ip));
038        allIps.add(ip);
039    }
040
041    public synchronized ClientBinding remove(final DuidOption.Duid clientId, final int iaid) {
042        if(bindings.containsKey(clientId)) {
043            final ClientBinding removed = bindings.get(clientId).remove(iaid);
044            allIps.remove(removed.getIp());
045            return removed;
046        }
047        return null;
048    }
049
050    public boolean containsIp(final InetAddress ip) {
051        return allIps.contains(ip);
052    }
053
054    public boolean contains(final DuidOption.Duid clientId, final int iaid) {
055        return bindings.containsKey(clientId) && bindings.get(clientId).contains(iaid);
056    }
057
058    @Nullable
059    public ClientBinding get(final DuidOption.Duid clientId, final int iaid) {
060        return bindings.get(clientId) == null ? null : bindings.get(clientId).get(iaid);
061    }
062
063    @Override public String toString() {
064        final StringBuilder sb = new StringBuilder("ClientBindingRegistry{");
065        sb.append("id='").append(id).append('\'');
066        sb.append('}');
067        return sb.toString();
068    }
069
070    public static ClientBindingRegistry createForIaNa() {
071        return new ClientBindingRegistry("IaNa");
072    }
073
074    public static ClientBindingRegistry createForIaTa() {
075        return new ClientBindingRegistry("IaTa");
076    }
077
078    public static class ClientBinding {
079
080        private final int iaId;
081        private InetAddress ip;
082
083        public ClientBinding(final int iaId, @Nonnull final InetAddress ip) {
084            this.iaId = iaId;
085            this.ip = ip;
086        }
087
088        public int getIaId() {
089            return iaId;
090        }
091
092        public InetAddress getIp() {
093            return ip;
094        }
095
096        @Override public String toString() {
097            return "ClientBinding{" +
098                "iaId=" + iaId +
099                ", ip=" + ip +
100                '}';
101        }
102
103        @Override public boolean equals(final Object o) {
104            if (this == o)
105                return true;
106            if (o == null || getClass() != o.getClass())
107                return false;
108            final ClientBinding that = (ClientBinding) o;
109            return Objects.equals(iaId, that.iaId);
110        }
111
112        @Override public int hashCode() {
113            return Objects.hash(iaId);
114        }
115    }
116
117    private static class ClientBindings {
118        private final Map<Integer, ClientBinding> bindings;
119
120        private ClientBindings() {
121            this.bindings = new HashMap<>();
122        }
123
124        ClientBinding get(int iaid) {
125            return bindings.get(iaid);
126        }
127
128        void add(ClientBinding binding) {
129            bindings.put(binding.iaId, binding);
130        }
131
132        ClientBinding remove(int iaid) {
133            return bindings.remove(iaid);
134        }
135
136        boolean contains(final int iaid) {
137            return bindings.containsKey(iaid);
138        }
139
140    }
141}
