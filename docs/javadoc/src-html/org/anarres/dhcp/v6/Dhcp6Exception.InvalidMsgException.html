021@Beta
022public class Dhcp6Exception extends DhcpException {
023
024    public Dhcp6Exception() {
025        super();
026    }
027
028    public Dhcp6Exception(@Nonnull final String description) {
029        super(description);
030    }
031
032    public Dhcp6Exception(@Nonnull final String description, @Nonnull final Exception e) {
033        super(description, e);
034    }
035
036    public static final class UnknownMsgException extends Dhcp6Exception {
037
038        private byte msgType;
039
040        public UnknownMsgException(byte msgType) {
041            super("Unrecognised msg type " + msgType);
042            this.msgType = msgType;
043        }
044
045        public byte getMsgType() {
046            return msgType;
047        }
048
049        /**
050         *
051         * Check if the message type is equal to DHCP_UNRECOGNIZED
052         *
053         * @throws UnknownMsgException in case the message is of unknown type
054         */
055        public static void check(@Nonnull final Dhcp6MessageType dhcp6MessageType, final byte type) throws UnknownMsgException {
056            if(dhcp6MessageType == Dhcp6MessageType.DHCP_UNRECOGNIZED) {
057                throw new UnknownMsgException(type);
058            }
059        }
060    }
061
062    public static final class InvalidMsgException extends Dhcp6Exception {
063
064        public InvalidMsgException(@Nonnull final String message) {
065            super(message);
066        }
067
068        public InvalidMsgException(@Nonnull final Dhcp6MessageType msgType) {
069            this("Invalid DHCP message: " + msgType);
070        }
071
072        /**
073         * Check parsed DHCP_SOLICIT message for correct type and options
074         *
075         * https://tools.ietf.org/html/rfc3315#section-15.2
076         *
077         * @throws org.anarres.dhcp.v6.Dhcp6Exception.InvalidMsgException in case the message is invalid and should be discarded
078         */
079        public static void checkSolicit(@Nonnull final Dhcp6Message msg) throws InvalidMsgException {
080            checkMsgType(msg, Dhcp6MessageType.DHCP_SOLICIT);
081            checkOption(msg, ClientIdOption.class);
082            if(hasOption(msg, ServerIdOption.class)) {
083                throw new InvalidMsgException(String.format("Unexpected option: %s", ServerIdOption.class));
084            }
085        }
086
087        private static void checkOption(@Nonnull final Dhcp6Message msg, @Nonnull final Class<? extends Dhcp6Option> type) throws InvalidMsgException {
088            if(noOption(msg, type)) {
089                throw new InvalidMsgException(String.format("Missing option: %s", type));
090            }
091        }
092
093        private static void checkNoOption(@Nonnull final Dhcp6Message msg, @Nonnull final Class<? extends Dhcp6Option> type) throws InvalidMsgException {
094            if(!noOption(msg, type)) {
095                throw new InvalidMsgException(String.format("Unexpected option: %s", type));
096            }
097        }
098
099        private static void checkMsgType(@Nonnull final Dhcp6Message msg, @Nonnull final Dhcp6MessageType expected) throws InvalidMsgException {
100            if(!isMsgType(msg, expected)) {
101                throw new InvalidMsgException(String.format("Incorrect type, expected: %s, but was: %s", expected, msg.getMessageType()));
102            }
103        }
104
105        private static boolean noOption(@Nonnull final Dhcp6Message msg, @Nonnull final Class<? extends Dhcp6Option> type) {
106            return msg.getOptions().get(type) == null;
107        }
108
109        private static boolean hasOption(@Nonnull final Dhcp6Message msg, @Nonnull final Class<? extends Dhcp6Option> type) {
110            return msg.getOptions().get(type) != null;
111        }
112
113        private static boolean isMsgType(@Nonnull final Dhcp6Message msg, @Nonnull final Dhcp6MessageType type) {
114            return msg.getMessageType() == type;
115        }
116
117        /**
118         * Check parsed DHCP_REQUEST message for correct type and options
119         *
120         * https://tools.ietf.org/html/rfc3315#section-15.4
121         *
122         * @throws org.anarres.dhcp.v6.Dhcp6Exception.InvalidMsgException in case the message is invalid and should be discarded
123         */
124        public static void checkRequest(@Nonnull final Dhcp6Message msg, @Nonnull final DuidOption.Duid duid) throws InvalidMsgException {
125            checkMsgType(msg, Dhcp6MessageType.DHCP_REQUEST);
126            checkOption(msg, ServerIdOption.class);
127            checkOptionValue(msg, ServerIdOption.class, duid.getData());
128            checkOption(msg, ClientIdOption.class);
129        }
130
131        private static void checkOptionValue(@Nonnull final Dhcp6Message msg,
132            @Nonnull final Class<? extends Dhcp6Option> optionType, @Nonnull final byte[] expectedValue)
133            throws InvalidMsgException {
134            if (!isOptionValue(msg, optionType, expectedValue)) {
135                throw new InvalidMsgException(String
136                    .format("Unexpected option value: %s, should be: %s", ServerIdOption.class,
137                        Arrays.toString(expectedValue)));
138            }
139        }
140
141        private static boolean isOptionValue(@Nonnull final Dhcp6Message msg, @Nonnull final Class<? extends Dhcp6Option> optionType,
142            @Nonnull final byte[] expectedValue) {
143            final Dhcp6Option dhcp6Option = Preconditions.checkNotNull(msg.getOptions().get(optionType), "Option %s missing within: %s", optionType, msg.getOptions());
144            return Arrays.equals(dhcp6Option.getData(), expectedValue);
145        }
146
147        /**
148         * Check parsed DHCP_REQUEST message for correct type and options
149         *
150         * https://tools.ietf.org/html/rfc3315#section-15.9
151         *
152         * @throws org.anarres.dhcp.v6.Dhcp6Exception.InvalidMsgException in case the message is invalid and should be discarded
153         */
154        public static void checkRelease(final Dhcp6Message msg, final DuidOption.Duid duid) throws InvalidMsgException {
155            checkMsgType(msg, Dhcp6MessageType.DHCP_RELEASE);
156            checkOption(msg, ServerIdOption.class);
157            checkOptionValue(msg, ServerIdOption.class, duid.getData());
158            checkOption(msg, ClientIdOption.class);
159        }
160
161        /**
162         * https://tools.ietf.org/html/rfc3315#section-7.1
163         */
164        public static void checkRelayForward(@Nonnull final Dhcp6Message msg) throws InvalidMsgException {
165            checkMsgType(msg, Dhcp6MessageType.DHCP_RELAY_FORW);
166            checkOption(msg, RelayMessageOption.class);
167        }
168
169        /**
170         *
171         * https://tools.ietf.org/html/rfc3315#section-15.12
172         */
173        public static void checkInformationRequest(final Dhcp6Message msg, final DuidOption.Duid duid)
174            throws InvalidMsgException {
175            checkMsgType(msg, Dhcp6MessageType.DHCP_INFORMATION_REQUEST);
176            if(msg.getOptions().contains(ServerIdOption.class)) {
177                checkOptionValue(msg, ServerIdOption.class, duid.getData());
178            }
179            checkNoOption(msg, IaNaOption.class);
180            checkNoOption(msg, IaTaOption.class);
181        }
182
183        /**
184         * https://tools.ietf.org/html/rfc3315#section-15.5
185         */
186        public static void checkConfirm(final Dhcp6Message msg) throws InvalidMsgException {
187            checkMsgType(msg, Dhcp6MessageType.DHCP_CONFIRM);
188            checkOption(msg, ClientIdOption.class);
189            checkNoOption(msg, ServerIdOption.class);
190        }
191
192        /**
193         * https://tools.ietf.org/html/rfc3315#section-15.7
194         */
195        public static void checkRebind(final Dhcp6Message msg) throws InvalidMsgException {
196            checkMsgType(msg, Dhcp6MessageType.DHCP_REBIND);
197            checkNoOption(msg, ServerIdOption.class);
198            checkOption(msg, ClientIdOption.class);
199        }
200
201        /**
202         * https://tools.ietf.org/html/rfc3315#section-15.6
203         */
204        public static void checkRenew(final Dhcp6Message msg, final DuidOption.Duid duid) throws InvalidMsgException {
205            checkMsgType(msg, Dhcp6MessageType.DHCP_RENEW);
206            checkOption(msg, ServerIdOption.class);
207            checkOptionValue(msg, ServerIdOption.class, duid.getData());
208            checkOption(msg, ClientIdOption.class);
209        }
210
211        /**
212         * https://tools.ietf.org/html/rfc3315#section-15.8
213         */
214        public static void checkDecline(final Dhcp6Message msg, final DuidOption.Duid duid) throws InvalidMsgException {
215            checkMsgType(msg, Dhcp6MessageType.DHCP_RENEW);
216            checkOption(msg, ServerIdOption.class);
217            checkOptionValue(msg, ServerIdOption.class, duid.getData());
218            checkOption(msg, ClientIdOption.class);
219        }
220    }

226

228            super();
229        }
230
231        public UnableToAnswerException(@Nonnull final String description) {
232            super(description);
233        }

235        public UnableToAnswerException(@Nonnull final String description, @Nonnull final Exception e) {
236            super(description, e);
237        }
238    }
